program gift1;
type tree=^treetype;
     treetype=record
       na:string;
       money:longint;
       lchild,rchild:tree;
     end;
var n,i,j,k,code,ngi,m:longint;
    root,p,q,pre:tree;
    st:string;
    a:array[1..10000] of tree;
procedure init;
  begin
    readln(st);
    new(root);
    root^.na:=st;
    root^.money:=0;
    root^.lchild:=NIL;
    root^.rchild:=NIL;
    a[1]:=root;
  end;
begin
  assign(input,'gift1.in');assign(output,'gift1.out');
  reset(input);rewrite(output);
  readln(n);
  init;
  for i:=2 to n do
    begin
      readln(st);
      p:=root;
      pre:=p;
      while p<>NIL do
        begin
          if st=p^.na then break
            else if st<p^.na then
              begin
                code:=0;
                pre:=p;
                p:=p^.lchild;
              end
              else
                begin
                  code:=1;
                  pre:=p;
                  p:=p^.rchild;
                end;
        end;
      if p=NIL then
        begin
          new(p);
          p^.na:=st;
          p^.money:=0;
          p^.lchild:=NIL;
          p^.rchild:=NIL;
          if code=0 then pre^.lchild:=p
            else pre^.rchild:=p;
          a[i]:=p;
        end;
    end;
  while not(eof(input)) do
    begin
      readln(st);
      readln(m,ngi);
      p:=root;
      while p^.na<>st do
        begin
          if st<p^.na then p:=p^.lchild
            else p:=p^.rchild;
        end;
      if ngi>0 then p^.money:=p^.money+m mod ngi
        else p^.money:=p^.money+m;
      p^.money:=p^.money-m;
      for i:=1 to ngi do
        begin
          readln(st);
          p:=root;
          while p^.na<>st do
            begin
              if st<p^.na then p:=p^.lchild
                else p:=p^.rchild;
            end;
          inc(p^.money,m div ngi);
        end;
    end;
  for i:=1 to n do
    writeln(a[i]^.na,' ',a[i]^.money);
  close(input);
  close(output);
end.
