int main()
{
	int i,j;
	int k;
	int temp;
	FILE *fin,*fout;
	fin=fopen("castle.in","r");
	fout=fopen("castle.out","w");
	fscanf(fin,"%d%d",&M,&N);
	for(i=0;i<N;i++)
	{
		for(j=0;j<M;j++)
		{
			fscanf(fin,"%d",&temp);
			for(k=3;k>=0;k--)
			{
				castle[i][j][k]=temp%2;
				temp/=2;
			}
			castle[i][j][4]=0;
		}
	}
	mark[0]=1;
	for(i=0;i<N;i++)
	{
		for(j=0;j<M;j++)
		{
			if(castle[i][j][4]==0)
			{
				mark[mark[0]]=0;
				Markhome(i,j,4);
				mark[0]++;
			}
		}
	}
 
	fprintf(fout,"%d\n",mark[0]-1);
	j=0;
	for(i=1;i<mark[0];i++)
	{
		if(mark[i]>j) j=mark[i];
	}
	fprintf(fout,"%d\n",j);
	for(j=0;j<M;j++)
	{
		for(i=N-1;i>=0;i--)
		{
			Addhome(i,j);
		}
	}
	fprintf(fout,"%d\n",max);
	fprintf(fout,"%d %d %c\n",wall[0]+1,wall[1]+1,wall[2]);
	return 0;
}
int Markhome(int x,int y,int from)
{
	castle[x][y][4]=mark[0];
	mark[mark[0]]++;
	if(castle[x][y][0]==0 && castle[x+1][y][4]==0 && x<N-1) Markhome(x+1,y,2);
	if(castle[x][y][1]==0 && castle[x][y+1][4]==0 && y<M-1) Markhome(x,y+1,3);
	if(castle[x][y][2]==0 && castle[x-1][y][4]==0 && x!=0) Markhome(x-1,y,0);
	if(castle[x][y][3]==0 && castle[x][y-1][4]==0 && y!=0) Markhome(x,y-1,1);
	return 0;
}
int Addhome(int x,int y)
{
	int temp;
	if(x>0 && castle[x-1][y][4]!=castle[x][y][4])
	{
		temp=mark[castle[x-1][y][4]]+mark[castle[x][y][4]];
		if(max<temp)
		{
			max=temp;
			wall[0]=x;
			wall[1]=y;
			wall[2]='N';
		}
	}
	if(y<M-1 && castle[x][y+1][4]!=castle[x][y][4]) 
	{
		temp=mark[castle[x][y+1][4]]+mark[castle[x][y][4]];
		if(max<temp)
		{
			max=temp;
			wall[0]=x;
			wall[1]=y;
			wall[2]='E';
		}
	}
	return 0;
}

#define MAXDIM 50
#define MAXN 100

#define MAXROOM (MAXDIM*MAXDIM)
 
enum {
    Wwest = 1,
    Wnorth = 2,
    Weast = 4,
    Wsouth = 8
};
 
typedef struct Square	Square;
struct Square {
    int wall;
    int numbered;
    int room;
};
 
int wid, ht;
Square castle[MAXDIM][MAXDIM];
int roomsize[MAXROOM];
 
void
number(int room, int x, int y)
{
    int w;
 
    if(castle[x][y].numbered) {
	assert(castle[x][y].room == room);
	return;
    }
 
    castle[x][y].numbered = 1;
    castle[x][y].room = room;
    roomsize[room]++;
 
    w = castle[x][y].wall;
 
    if(x > 0 && !(w & Wwest))
	number(room, x-1, y);
 
    if(x+1 < wid && !(w & Weast))
	number(room, x+1, y);
 
    if(y > 0 && !(w & Wnorth))
	number(room, x, y-1);
 
    if(y+1 < ht && !(w & Wsouth))
	number(room, x, y+1);
}
 
void
main(void)
{
    FILE *fin, *fout;
    int x, y, w, nroom, bigroom;
    int i, n, m, mx, my;
    char mc;
 
    fin = fopen("castle.in", "r");
    fout = fopen("castle.out", "w");
    assert(fin != NULL && fout != NULL);
 
    fscanf(fin, "%d %d", &wid, &ht);
 
    /* read in wall info */
    for(y=0; y<ht; y++) {
	for(x=0; x<wid; x++) {
	    fscanf(fin, "%d", &w);
	    castle[x][y].wall = w;
	}
    }
 
    /* number rooms */
    nroom = 0;
    for(y=0; y<ht; y++)
    for(x=0; x<wid; x++)
	if(!castle[x][y].numbered)
	    number(nroom++, x, y);
 
    /* find biggest room */
    bigroom = roomsize[0];
    for(i=1; i<nroom; i++)
	if(bigroom < roomsize[i])
	    bigroom = roomsize[i];
 
    /* look at best that can come of removing an east or north wall */
    m = 0;
    for(x=0; x<wid; x++)
    for(y=ht-1; y>=0; y--) {
	if(y > 0 && castle[x][y].room != castle[x][y-1].room) {
	    n = roomsize[castle[x][y].room] + roomsize[castle[x][y-1].room];
	    if(n > m) {
		m = n;
		mx = x;
		my = y;
		mc = 'N';
	    }
	}
	if(x+1 < wid && castle[x][y].room != castle[x+1][y].room) {
	    n = roomsize[castle[x][y].room] + roomsize[castle[x+1][y].room];
	    if(n > m) {
		m = n;
		mx = x;
		my = y;
		mc = 'E';
	    }
	}
    }
 
    fprintf(fout, "%d\n", nroom);
    fprintf(fout, "%d\n", bigroom);
    fprintf(fout, "%d\n", m);
    fprintf(fout, "%d %d %c\n", my+1, mx+1, mc);
    exit(0);
}

#include <stdio.h>
 
#define IWW(m)	( m & 1 )
#define IWN(m)	( m & 2 )
#define	IWE(m)	( m & 4 )
#define IWS(m)	( m & 8 )
#define MAX(a, b)	( ((a)>(b))?(a):(b) )
 
#define INPUT	"castle.in"
#define OUTPUT	"castle.out"
 
int N = 0, M = 0;		/* row and col */
int walls[51][51] = {{0}};
int color[51][51] = {{0}};
int nroom[2501] = {0};
int maxnroom = 0;
int maxnroombro = 0;
int removingwally = 0, removingwallx = 0, removingwalld = 0;
 
void
floodfill(int y, int x, int co, int * counter)
{
	if( color[y][x] )
		return;
 
	(*counter)++;
	color[y][x] = co;
	if( 0 == IWW(walls[y][x]) )	floodfill(y, x-1, co, counter);
	if( 0 == IWN(walls[y][x]) )	floodfill(y-1, x, co, counter);
	if( 0 == IWE(walls[y][x]) )	floodfill(y, x+1, co, counter);
	if( 0 == IWS(walls[y][x]) )	floodfill(y+1, x, co, counter);
}
 
void
testremoving(int y, int x, int d)
{
	switch(d)
	{
	case 'N':
		if( nroom[color[y][x]-1] + nroom[color[y-1][x]-1] > maxnroombro  )
		{
			removingwally = y;
			removingwallx = x;
			removingwalld = d;
			maxnroombro = nroom[color[y][x]-1] + nroom[color[y-1][x]-1];
		}
		break;
	case 'E':
		if( nroom[color[y][x]-1] + nroom[color[y][x+1]-1] > maxnroombro )
		{
			removingwally = y;
			removingwallx = x;
			removingwalld = d;
			maxnroombro = nroom[color[y][x]-1] + nroom[color[y][x+1]-1];
		}
		break;
	}
}
