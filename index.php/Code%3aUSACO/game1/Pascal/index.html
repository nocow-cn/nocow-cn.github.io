const

var
  n,i,p,j,k:longint;
  a,sum:array[0..maxn]of longint;
  s:array[1..maxn,1..maxn]of longint;
  f:array[1..maxn,1..maxn]of longint;
begin
  assign(input,'game1.in'); reset(input);
  assign(output,'game1.out'); rewrite(output);
  readln(n);
  for i:=1 to n do
  begin
    read(a[i]);
    sum[i]:=sum[i-1]+a[i];
  end;
  for i:=1 to n do
    for j:=i to n do
      s[i,j]:=sum[j]-sum[i-1];
  for i:=1 to n do f[i,i]:=a[i];
  for p:=1 to n-1 do
    for i:=1 to n-p+1 do
    begin
      j:=i+p;
      if a[i]+s[i+1,j]-f[i+1,j]>a[j]+s[i,j-1]-f[i,j-1]
        then f[i,j]:=a[i]+s[i+1,j]-f[i+1,j]
        else f[i,j]:=a[j]+s[i,j-1]-f[i,j-1];
    end;
  writeln(f[1,n],' ',s[1,n]-f[1,n]);
  close(input); close(output);
end.

    sum,dp:array[1..100,1..100] of longint;
    n,i,l,j,k:longint;
function min(a,b:longint):longint;
  begin
    if a>b then exit(b)
      else exit(a);
  end;
begin
  assign(input,'game1.in');assign(output,'game1.out');
  reset(input);rewrite(output);
  readln(n);
  for i:=1 to n do
    read(a[i]);
  for i:=1 to n do
    sum[i,1]:=a[i];
  for i:=1 to n do
    for l:=2 to n-i+1 do
      sum[i,l]:=sum[i,l-1]+a[i+l-1];
  for i:=1 to n do
    dp[i,1]:=a[i];
  for l:=2 to n do
    for i:=1 to n-l+1 do
      dp[i,l]:=sum[i,l]-min(dp[i,l-1],dp[i+1,l-1]);
  writeln(dp[1,n],' ',sum[i,n]-dp[i,n]);
  close(input);
  close(output);

         else exit(a);
     end;
 
 procedure dp;
    var i,j,t:longint;
   begin
     t:=0;
       while t<n do
         begin
           for i:=1 to n-t do
              begin
                j:=i+t;
                f[i,j]:=sum[i,j]-min(f[i,j-1],f[i+1,j]);
              end;
            inc(t);
         end;
    end;
 
   begin

  begin

}var
  n,i,j,l,r:longint;
  a:array[1..100]of longint;
  w,q:array[1..100,1..100]of longint;
  p:array[1..2]of longint;
procedure fly(o:longint);
var
  tmp,ans,tmp1,ans1:longint;
begin
    if w[l,r]<>0 then
      begin
       p[o]:=p[o]+w[l,r];
       p[o mod 2+1]:=p[o mod 2+1]+q[l,r];
       exit;
      end;
    if l=r then p[o]:=p[o]+a[l]
    else
    begin
    tmp:=p[o];
    tmp1:=p[o mod 2+1];
    p[o]:=p[o]+a[l];
    inc(l);
    fly(o mod 2+1);
    ans:=p[o];
    ans1:=p[o mod 2+1];
    p[o]:=tmp+a[r];
    p[o mod 2+1]:=tmp1;
    dec(l);
    dec(r);
    fly(o mod 2+1);
    if p[o]<ans then
      begin
        p[o]:=ans;
        p[o mod 2+1]:=ans1;
      end;
       inc(r);
      w[l,r]:=p[o]-tmp;
      q[l,r]:=p[o mod 2+1]-tmp1;
    end;
end;
begin
   assign(input,'game1.in');reset(input);
   assign(output,'game1.out');rewrite(output);
   readln(n);
   for i:=1 to n do
     read(a[i]);
     l:=1;
     r:=n;
   fly(1);
   writeln(p[1],' ',p[2]);
   close(input);
   close(output);
end.
