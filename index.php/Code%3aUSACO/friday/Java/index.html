import java.io.*;
 
public class friday {
	static boolean isLeap(int year) {
		if (year % 100 == 0) {
			year /= 100;
		}
		return (year % 4 == 0) ? true : false;
	}
	public static void countOneYear(int year, int[] count) {
		int weekOfone = 1;
		int tempYear = year;
		while (tempYear > 1900) {
			weekOfone++;
			if (isLeap(tempYear))
				weekOfone++;
			if (weekOfone > 7)
				weekOfone %= 7;
			tempYear--;
		}
		boolean leap = isLeap(year);
		if (leap)
			weekOfone--;
		int day = 13 - 31 + weekOfone - 8;
 
		for (int i = 0; i < 12; i++) {
			if (i == 4 || i == 9 || i == 6 || i == 11) {
				day += 30;
			} else if (i == 2) {
				if (leap)
					day += 29;
				else
					day += 28;
			} else {
				day += 31;
			}
			count[(day) % 7] += 1;
		}
	}
