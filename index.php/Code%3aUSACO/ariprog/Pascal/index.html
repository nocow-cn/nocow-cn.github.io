     if max<c[i]+(n-1)*k then break;
     while (t<n)and(s<=max) do
      begin
        s:=s+k;
        if not p[s] then break;
        inc(t);
      end;
     if t=n then
      begin
        inc(l);
        a[l,1]:=c[i];
        a[l,2]:=k;
      end;
   end;
 for i:=1 to l do writeln(a[i,1],' ',a[i,2]);
 if l=0 then writeln('NONE');
 close(input);  close(output);

end.
