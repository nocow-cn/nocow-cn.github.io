#include <iostream>
 
using namespace std;
 
struct milk
{
    int total,present = 0;
}m[3];
 
bool check[21][21][21] = {false};
 
void pour(milk &a, milk &b)
{
    if (a.present > b.total - b.present) a.present -= b.total - b.present, b.present = b.total;
    else b.present += a.present, a.present = 0;
}
 
void pro(int a, int b, int c)
{
    int i,j;
    m[0].present = a, m[1].present = b, m[2].present = c;
    for(i=0;i<3;i++)
        for(j=0;j<3;j++)
        {
            if(j==i) j++;
            else
            {
                pour(m[i],m[j]);
                if(!check[m[0].present][m[1].present][m[2].present])
                {
                    check[m[0].present][m[1].present][m[2].present] = true;
                    pro(m[0].present, m[1].present, m[2].present);
                }
                m[0].present = a, m[1].present = b, m[2].present = c;
            }
        }
}
