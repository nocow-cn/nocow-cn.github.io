#include <stdio.h>
#include <string.h>

int h,w,n;
char map[31][31];
int z[26][4];

int v[26];
int s[26];
char x[26];
int an;
char ans[10000][26];

void swap(int i,int j){
	char y[26];
	strcpy(y,ans[i]);
	strcpy(ans[i],ans[j]);
	strcpy(ans[j],y);
}

void P(int p,int r){
	int i=p,j=r+1;
	if(p>=r)return;
	swap(p,(p+r)/2);
	while(1){
		while(strcmp(ans[++i],ans[p])<0 && i<=r);
		while(strcmp(ans[p],ans[--j])<0 && j>=p);
		if(i>=j)break;
		swap(i,j);
	}
	swap(p,j);
	P(p,j-1);
	P(j+1,r);
}

void search(int q){
	int i,j,k;
	if(q==-1){
		strcpy(ans[an],x);
		an++;
		return;
	}
	char ma[31][31];
	for(j=0;j<h;j++)
		for(k=0;k<w;k++)
			ma[j][k]=map[j][k];
	for(i=25;i>=0;i--){
		if(v[i]==0 || s[i]==1)continue;
		for(j=z[i][0];j<=z[i][2];j++)
			if(map[j][z[i][1]]!=i+'A' && map[j][z[i][1]]!='?')break;
			else if(map[j][z[i][3]]!=i+'A' && map[j][z[i][3]]!='?')break;
		if(j!=z[i][2]+1)continue;
		for(j=z[i][1];j<=z[i][3];j++)
			if(map[z[i][0]][j]!=i+'A' && map[z[i][0]][j]!='?')break;
			else if(map[z[i][2]][j]!=i+'A' && map[z[i][2]][j]!='?')break;
		if(j!=z[i][3]+1)continue;
		x[q]=i+'A';
		s[i]=1;
		for(j=z[i][0];j<=z[i][2];j++)
			map[j][z[i][1]]=map[j][z[i][3]]='?';
		for(j=z[i][1];j<=z[i][3];j++)
			map[z[i][0]][j]=map[z[i][2]][j]='?';
		search(q-1);
		for(j=0;j<h;j++)
			for(k=0;k<w;k++)
				map[j][k]=ma[j][k];
		s[i]=0;
	}
}

main(){
	freopen("frameup.in","r",stdin);
	freopen("frameup.out","w",stdout);
	int i,j;
	scanf("%d%d",&h,&w);
	for(i=0;i<h;i++)scanf("%s",map[i]);
	for(i=0;i<26;i++)
		z[i][0]=z[i][1]=40;
	for(i=0;i<h;i++)
		for(j=0;j<w;j++)
			if(map[i][j]!='.'){
				if(v[map[i][j]-'A']==0){v[map[i][j]-'A']=1;n++;}
				if(z[map[i][j]-'A'][0]>i)z[map[i][j]-'A'][0]=i;
				if(z[map[i][j]-'A'][1]>j)z[map[i][j]-'A'][1]=j;
				if(z[map[i][j]-'A'][2]<i)z[map[i][j]-'A'][2]=i;
				if(z[map[i][j]-'A'][3]<j)z[map[i][j]-'A'][3]=j;
			}
	search(n-1);
	P(0,an-1);
	for(i=0;i<an;i++)
		printf("%s\n",ans[i]);
	return 0;
}
