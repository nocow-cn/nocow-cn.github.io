program frameup;
  type
    ty=array[1..30]of char;
  var
    l,r,u,d:array['A'..'Z']of integer;{存储字母的上下左右边界}
    bb:array['A'..'Z','A'..'Z']of boolean;{字母之间的联系}
    du:array['A'..'Z']of longint;{字母的入度}
    w:array[1..100,1..100]of char;{读入数据}
    have:array['A'..'Z']of boolean;{字母是否出现过}
    used:array[1..26]of boolean;{拓扑排序时回溯时用到}
    ch:array[1..26]of char;{将出现的字母存入数组中}
    n,m,n1,len:longint;
    aa:array[1..10000]of string[30];{存储结果}
    a1:ty;
    ch1:char;
  procedure  int;
    var i,j:integer;
    begin
      assign(input,'frameup.in');reset(input);
      for ch1:='A' to 'Z' do
        begin
          l[ch1]:=100;r[ch1]:=-100;u[ch1]:=100;d[ch1]:=-100;
        end;
      readln(n,m);
      for i:=1 to n do
        begin
          for j:=1 to m do
           begin
            read(w[i,j]); ch1:=w[i,j];
            if (ord(w[i,j])>=ord('A'))and(ord(w[i,j])<=ord('Z')) then begin{计算上下左右边界}
            if j<l[w[i,j]] then l[w[i,j]]:=j;
            if j>r[w[i,j]] then r[w[i,j]]:=j;
            if i<u[w[i,j]] then u[w[i,j]]:=i;
            if i>d[w[i,j]] then d[w[i,j]]:=i;
            if have[w[i,j]]=false then
              begin
                have[w[i,j]]:=true;
                inc(n1);ch[n1]:=w[i,j];{将出现的字母存入ch数组}
              end; end;
           end;
          readln;
        end;
    end;
  procedure  try1(step:integer);{拓扑回溯过程}
    var
      i,j:integer;
      ch3:char;
    begin
      if step>n1 then begin inc(len);aa[len]:=a1;exit;end;
      for i:=1 to n1 do
        if (used[i]=false)and(du[ch[i]]=0) then
          begin
            ch3:=ch[i];
            for j:=1 to n1 do
              if bb[ch3,ch[j]]=true then dec(du[ch[j]]);
            a1[n1-step+1]:=ch3; used[i]:=true;
            try1(step+1);
            used[i]:=false;
            for j:=1 to n1 do
              if bb[ch3,ch[j]]=true then inc(du[ch[j]]);
          end;
    end;
  procedure  main;{先建立图形，再进行拓扑}
    var
      i,j:integer;
    begin
      for i:=1 to n1 do{建立图形的主过程}
        begin
          ch1:=ch[i];
          for j:=l[ch1] to r[ch1] do{扫描上下边界}
           begin
            if (w[u[ch1],j]<>ch1)and(w[u[ch1],j]<>'.') then{扫描上边界}
              if bb[w[u[ch1],j],ch1]=false then
                begin
                  inc(du[ch1]);
                  bb[w[u[ch1],j],ch1]:=true;
                end;
            if (w[d[ch1],j]<>ch1)and(w[d[ch1],j]<>'.') then{扫描下边界}
              if bb[w[d[ch1],j],ch1]=false then
                begin
                  inc(du[ch1]);
                  bb[w[d[ch1],j],ch1]:=true;
                end;
            end;
          for j:=u[ch1] to d[ch1] do{扫描左右边界}
            begin
              if (w[j,l[ch1]]<>ch1)and(w[j,l[ch1]]<>'.') then{扫描左边界}
                if bb[w[j,l[ch1]],ch1]=false then
                  begin
                    inc(du[ch1]);
                    bb[w[j,l[ch1]],ch1]:=true;
                  end;
              if (w[j,r[ch1]]<>ch1)and(w[j,r[ch1]]<>'.') then{扫描右边界}
                if bb[w[j,r[ch1]],ch1]=false then
                  begin
                    inc(du[ch1]);
                    bb[w[j,r[ch1]],ch1]:=true;
                  end;
            end;
        end;
      try1(1);
    end;

          while (i<j)and(aa[j]>temp) do dec(j);
          aa[i]:=aa[j];
          while (i<j)and(aa[i]<temp) do inc(i);
          aa[j]:=aa[i];
        end;
      aa[i]:=temp;
      k:=i;
    end;
  procedure  pai(s,t:longint);
    var k:longint;
    begin
      if s<t then
        begin
          kuai(s,t,k);
          pai(s,k-1);
          pai(k+1,t);
        end;
    end;

          for j:=1 to n1 do
            write(aa[i][j]);
          writeln;
        end;
      close(output);
    end;

此题的思路是扫描每一个字母图形的上下左右边界，进行拓扑排序 

 （1） 矩形的每条边中，至少有一部分是可见的。

于是我们可以轻易的确定每个矩形的边界。然后就对两个矩形的交点进行判断： 注意 

 （1）字母不是仅5个，而是26个大写字母都有。
 （2）要用快排（但在vijos上用一般排序就能0ms，因为vijos上的测试数据比较弱）

    mid,temp:string[maxH];

program frameup_YoungBoy;
const
  infile='frameup.in';
  outfile='frameup.out';
 
var
  map:array ['A'..'Z','A'..'Z'] of boolean; {拓扑排序的图}
  num:array ['A'..'Z'] of longint;  (记录边的入度}
  alive:array ['A'..'Z'] of boolean; {记录哪些字母存在}
  graph:array [1..30] of string[30];  {图形}
  ans:array [1..10000] of string[30]; {答案}
  tot,h,w,ldep:longint; {tot是解的数目,ldep是出现的字母数}
 
function Min(int1,int2:longint):longint;
begin
  if int1<int2 then Min:=int1 else Min:=int2;
end;
 
function Max(int1,int2:longint):longint;
begin
  if int1>int2 then Max:=int1 else Max:=int2;
end;
 
procedure Init;
var i:longint;
begin
  assign(input,infile); reset(input);
  readln(h,w);
  for i:=1 to h do readln(graph[i]);
  close(input);
end;
 
procedure Print;
var i:longint;
begin
  assign(output,outfile); rewrite(output);
  for i:=1 to tot do writeln(ans[i]);
  close(output);
end;
 
procedure MakeMap; {通过原图形求出拓扑图}
var
  c:char;
  i,j,x1,x2,y1,y2:longint;
begin
  fillchar(alive,sizeof(alive),false);
  fillchar(map,sizeof(map),false);
  ldep:=0;
  for c:='A' to 'Z' do
  for i:=1 to h do
    if pos(c,graph[i])>0 then
    begin
      inc(ldep);
      alive[c]:=true;
      break;
    end;
  for c:='A' to 'Z' do
    if alive[c] then
    begin
      x1:=maxlongint; y1:=maxlongint;
      x2:=0; y2:=0;
      for i:=1 to h do
      for j:=1 to w do
        if graph[i,j]=c then
        begin
          x1:=Min(x1,i); y1:=Min(y1,j);
          x2:=Max(x2,i); y2:=Max(y2,j);
        end;
      for i:=x1 to x2 do
      begin
        if not(map[c,graph[i,y1]]) then
        begin
          map[c,graph[i,y1]]:=true;
          inc(num[graph[i,y1]]);
        end;
        if not(map[c,graph[i,y2]]) then
        begin
          map[c,graph[i,y2]]:=true;
          inc(num[graph[i,y2]]);
        end;
      end;
      for i:=y1 to y2 do
      begin
        if not(map[c,graph[x1,i]]) then
        begin
          map[c,graph[x1,i]]:=true;
          inc(num[graph[x1,i]]);
        end;
        if not(map[c,graph[x2,i]]) then
        begin
          map[c,graph[x2,i]]:=true;
          inc(num[graph[x2,i]]);
        end;
      end;
    end;
end;
 
procedure DFS(depth:longint;now:string); {这就是深搜啦}
var
  c,cc:char;
  tmp:array ['A'..'Z'] of longint;
begin
  if depth>ldep then
  begin
    inc(tot);
    ans[tot]:=now;
    exit;
  end;
  tmp:=num;
  for c:='A' to 'Z' do
    if num[c]=1 then
    begin
      num[c]:=0;
      for cc:='A' to 'Z' do
        if (c<>cc) and (map[c,cc]) then dec(num[cc]);
      DFS(depth+1,now+c);
      num:=tmp;
      num[c]:=1;
    end;
end;
 
procedure Main;
var
  i,j,k:longint;
  t:string[30];
begin
  MakeMap;
  DFS(1,'');
end;
 
begin
  Init;
  Main;
  Print;
end.
