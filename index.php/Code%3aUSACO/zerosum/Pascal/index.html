Var
  i,n,num,answer,opperation:longint;
Procedure calculate(s:string); //处理sequence,O(n)
  Begin
    answer:=0;  num:=0;  opperation:=1;
    For i:=1 to length(s) do
      Case s[i] of
        '1'..'9':num:=num*10+ord(s[i])-ord('0');
        '+','-':Begin
                  Case opperation of
                    1:inc(answer,num);
                    2:dec(answer,num)
                  End;
                  Case s[i] of
                    '+':opperation:=1;
                    '-':opperation:=2
                  End;
                  num:=0
                End
      End;
    If answer=0 then writeln(copy(s,1,length(s)-1))
  End;
Procedure sequence(num:char;s:string); //构造sequence,O(3^n)
  Begin
    If num=char(n+1+ord('0')) then
      Begin
        calculate(s+'+');  exit
      End;
    sequence(succ(num),s+' '+num); //从这里可以看到,没有必要开数组记录答案
    sequence(succ(num),s+'+'+num); //只要按照' ''+''-'的顺序添加运算符
    sequence(succ(num),s+'-'+num)  //就可以保证输出顺序按ASCII表
  End;

program zerosum(input,output);
  var
    a:array[1..9] of integer;
    ans:array[1..8] of char;
    l:set of char=['+','-'];
    n:longint;
      procedure init;
        var
          i:longint;
        begin
          assign(input,'zerosum.in');
          assign(output,'zerosum.out');
          reset(input);
          rewrite(output);
          readln(n);
          for i:=1 to n do a[i]:=i;
          close(input);
        end;
          function calculate:longint;
            var
              i:longint;
              m,h:longint;
              k:string;
              z:char;
            begin
              k:='1';
              for z:='2' to chr(ord('0')+n) do
                begin
                  if ans[ord(z)-ord('1')]<>' ' then k:=k+ans[ord(z)-ord('1')];
                  k:=k+z;
                end;
              k:=k+'+';
              for i:=1 to length(k) do
                if k[i] in l then
                  begin
                    val(copy(k,1,i-1),m);
                    z:=k[i];
                    delete(k,1,i);
                    break;
                  end;
              while k<>'' do
                begin
                  for i:=1 to length(k) do
                    if k[i] in l then
                      begin
                        val(copy(k,1,i-1),h);
                        case z of
                            '+':m:=m+h;
                            '-':m:=m-h;
                          end;
                        z:=k[i];
                        delete(k,1,i);
                        break;
                      end;
                end;
              exit(m);
            end;
      procedure dfs(x:longint);
        var i:longint;
        begin
          if x=n then
            begin
              if calculate=0 then
                begin
                  write(a[1]);
                  for i:=1 to n-1 do write(ans[i],a[i+1]);
                  writeln;
                end;
              exit;
            end;
          ans[x]:=' ';
          dfs(x+1);
          ans[x]:='+';
          dfs(x+1);
          ans[x]:='-';
          dfs(x+1);
        end;
  begin
    init;
    dfs(1);
