var
  t:char;
  n,i,j,dpx,tt:byte;
  dp:array[0..250]of byte;
  count:array[2..250]of word;
begin
  assign(input,'range.in');
  assign(output,'range.out');
  reset(input);
  rewrite(output);
  readln(n);
  for i:=1 to n do
  begin
    dp[0]:=0;
    for j:=1 to n do
    begin
      read(t);
      if t='0' then
      begin
        dpx:=dp[j];
        dp[j]:=0;
      end
      else
      begin
        tt:=255;
        if dp[j]<tt then
          tt:=dp[j];
        if dpx<tt then
          tt:=dpx;
        if dp[j-1]<tt then
          tt:=dp[j-1];
        dpx:=dp[j];
        dp[j]:=tt+1;
        if dp[j]>=2 then
          inc(count[dp[j]]);
      end;
    end;
    readln;
  end;
  for i:=n-1 downto 2 do
    inc(count[i],count[i+1]);
  for i:=2 to n do
    if count[i]=0 then
      break
    else
      writeln(i,' ',count[i]);
  close(output);
end.

//很不错的做法
program range;
const
  maxn=250;
var
  fin,fout:text;
  map:array[1..maxn,1..maxn]of boolean;
  n,size,i,j,count:longint;
  c:char;
begin
  assign(fin,'range.in');
  reset(fin);
  readln(fin,n);
  for i:=1 to n do begin
    for j:=1 to n do begin
      read(fin,c);
      if c='1' then map[i,j]:=true else map[i,j]:=false;
    end;
    readln(fin);
  end;
  close(fin);
 
  assign(fout,'range.out');
  rewrite(fout);
  for size:=2 to n do begin
    count:=0;
    for i:=1 to n+1-size do
      for j:=1 to n+1-size do begin
        map[i,j]:=map[i,j] and map[i,j+1] and map[i+1,j] and map[i+1,j+1];
        if map[i,j] then inc(count);
      end;
    if count>0 then writeln(fout,size,' ',count);
    if count<4 then break;
  end;
  close(fout);
end.

program range;
var
 f1,f2:text;
 n,e,i,k:longint;
 s:array[0..250]of string;
 b:array[0..250]of longint;
 a:array[0..250,0..250]of longint;
function min(x,y,z:longint):longint;
 begin
  min:=x;
  if y<min then
   min:=y;
  if z<min then
   min:=z;
 end;
begin
 assign(f1,'range.in');
 reset(f1);
 readln(f1,n);
 for e:=1 to n do
  readln(f1,s[e]);
 close(f1);
 for e:=1 to n do
  for i:=1 to n do
   if s[e][i]='1' then
    a[e,i]:=1;
 for e:=2 to n do
  for i:=2 to n do
   if a[e,i]<>0 then
    a[e,i]:=min(a[e-1,i-1],a[e,i-1],a[e-1,i])+1;
 for e:=1 to n do
  for i:=1 to n do
   if a[e,i]>1 then
    for k:=a[e,i] downto 2 do
     inc(b[k]);
 assign(f2,'range.out');
 rewrite(f2);
 for e:=1 to n do
  if b[e]<>0 then
   writeln(f2,e,' ',b[e]);
 close(f2);
end.

program range;
var
 f:array[1..250,1..250]of longint;
 i,j,k,t,s,n:longint;
 c:char;
begin
 assign(input,'range.in');reset(input);
 assign(output,'range.out');rewrite(output);
 readln(n);
 
 for i:=1 to n do
  begin
   for j:=1 to n do
    begin
     read(c);
     f[i,j]:=f[i-1,j]+f[i,j-1]-f[i-1,j-1];
     if c='1' then inc(f[i,j]);
    end;
   readln;
  end;
 
 for k:=2 to n do
  begin
   s:=k*k;t:=0;
   for i:=0 to n-k do for j:=0 to n-k do
    if f[i+k,j+k]+f[i,j]-f[i,j+k]-f[i+k,j]=s then inc(t);
   if t>0 then writeln(k,' ',t);
  end;
 close(output);
end.

虽然题目很简单，但是这个题目仍然给人以深刻的启示：那就是如何压缩存储量？
注意的上面诸位的算法中只有starfor的是控制在O(n)的空间复杂度的，虽然对
于250的数据量并没有节省空间的必要，但是如果在保证O(n^2)的时效下依然能
够大幅度地压缩空间，那么何乐而不为呢？
直接读进数据是造成空间浪费的原因，因此，不妨边读入，边做DP,这样，空间使
用变为原来的1/250

var map:array[0..1,0..251] of longint;
    up,left:array[0..251] of longint;
    f:array[0..251] of longint;
    i,j,n,t:longint;
    ch:char;
function min(a,b,c:longint):longint;
begin
 if a<b then b:=a;
 if b<c then exit(b) else exit(c);
end;
 
 
begin
 assign(input,'range.in');
 reset(input);
 assign(output,'range.out');
 rewrite(output);
 readln(n);
 t:=0;
 for i:=1 to n do
 begin
   for j:=1 to n do
   begin
   read(ch);
     map[t,j]:=ord(ch)-ord('0');
     if map[t,j]<>0 then begin left[j]:=left[j-1]+1;
                               up[j]:=up[j]+1;
                         end else
                         begin left[j]:=0;up[j]:=0;end;
     map[t,j]:=min(map[1-t,j-1]+1,up[j],left[j]);
     if map[t,j]<>0 then inc(f[map[t,j]]);
   end;
   readln;
   t:=1-t;
 end;
 for i:=n downto 1 do inc(f[i],f[i+1]);
 for i:=2 to n do if f[i]<>0 then writeln(i,' ',f[i])
                             else break;
close(output);
end.

var
  n,i,j:longint;
  x:char;
  a,opt:array[0..251,0..251] of longint;
  f:array[0..251] of longint;
 
procedure Init;
  begin
    readln(n);
    for i:=1 to n do
     begin
      for j:=1 to n do
        begin
          read(x);
          val(x,a[i,j]);
        end;
      readln;
     end;
  end;
 
function min(a,b,c:longint):longint;
  begin
    if a<b then min:=a else min:=b;
    if c<min then min:=c;
  end;
 
procedure Main;
  begin
    for i:=1 to n do
      for j:=1 to n do
        if a[i,j]=1 then
          opt[i,j]:=min(opt[i-1,j],opt[i,j-1],opt[i-1,j-1])+1;
    for i:=1 to n do
      for j:=1 to n do
        if opt[i,j]>=2 then
          inc(f[opt[i,j]]);
    for i:=1 to n do
      if f[i]<>0 then
        for j:=i+1 to n do
          inc(f[i],f[j]);
    for i:=1 to n do
      if f[i]<>0 then
        writeln(i,' ',f[i]);
    close(output);
  end;
 
 
begin
  assign(input,'range.in');
  assign(output,'range.out');
  reset(input);
  rewrite(output);
  Init;
  Main;
end.
