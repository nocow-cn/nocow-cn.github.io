using namespace std;
const int MAX = 1005;
int ar[MAX];//初始位置
int br[MAX];//正确位置
int n;//奖牌个数
int c[4][4]={0};//c[i][j] : 某个位置应该是i但却是j，这样的位置的个数
int csum=0;//等待交换的数的个数
int main() {
	pinit();
	cin >> n;
	for (int i = 0; i < n; ++ i) {
		cin >> ar[i];
		br[i] = ar[i];
	}
	sort(br, br+n);
	for (int i = 0; i < n; ++ i) {
		if (ar[i] != br[i]) {
			++c[br[i]][ar[i]];
			++csum;
		}
	}
	int ans = 0;
	for (int i = 1; i <= 3; ++ i) {
		for (int j = i+1; j <= 3; ++ j) {
			//如果有i->j，又有j->i，就该尽可能的先将它们交换，每对儿这样的i与j耗费1次交换
			int t = min(c[i][j], c[j][i]);
			ans += t;
			c[i][j] -= t;
			c[j][i] -= t;
			//待交换的数的个数减少了
			csum -= 2*t;
		}
	}
	//剩下待交换的数的个数肯定是3的倍数，即只会出现i->j j->k k->i，这种情况耗费2次交换
	ans += csum/3*2;
	cout << ans << endl;
	return 0;
}

#include <fstream>
using namespace std;
ifstream fin("sort3.in");
ofstream fout("sort3.out");
int main()
{
	int N,c12=0,c21=0,c31=0,c32=0;//cxy为在x应在位置（即排序后x的位置，下面简称为位x）上的y的个数
	int count[4]={0,0,0,0};
	fin>>N;
	int* p=new int[N];
	for (int i=0;i!=N;++i){
		fin>>p[i];
		++count[p[i]];
	}//读入数据，统计1，2，3的个数
	for (int i=0;i!=N;++i)
		if (i<count[1] && p[i]==2) 								++c12;
		else if (i>=count[1]+count[2] && p[i]!=3) p[i]==1?++c31:++c32;
		else if (count[1]<=i && i<count[1]+count[2] && p[i]==1) ++c21;//计算相应的cxy，只计算用到的
	fout<<c31+c32+(c21>c12?c21:c12)<<endl;/*采用数学方法计算出结果，以c12>c21为例（反过来只需把所有的1与2互换即可）
	首先把位2上的1与位1上的2互换位置，这以后位1中有2，位2中无1；然后把位3中的1全部交换回位1
	这以后位3中的2分别为其固有的2(c32)和从1中交换来的2(c12-c21)组成，再把它们交换回位2，化简后即为程序中的表达式*/
	return 0;
}
