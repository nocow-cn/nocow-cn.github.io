program         __SHUTTLE__ ;
var     n , ctrl: longint ;
procedure       dfs (i : longint) ;
var     st , o , p , j
                : longint ;
begin
        if (i and 1 = 1) then begin
         st := n + i ; o := -2 ;
        end
        else begin
         st := n - i + 2 ; o := 2 ;
        end ;
 
        p := st ;
        for j := 1 to i do begin
         if (ctrl >= 1) then begin
          if (ctrl mod 20 <> 1) then write (' ') ;
          write (p) ;
          if (ctrl mod 20 = 0) then
           writeln ;
         end ;
         inc (ctrl) ;
         inc (p , o) ;
        end ;
 
        if (i = n + 1) then exit ;
 
        dfs (i + 1) ;
 
        p := st ;
        for j := 1 to i do begin
         if (ctrl >= 1) then begin
          if (ctrl mod 20 <> 1) then write (' ') ;
          write (p) ;
          if (ctrl mod 20 = 0) then
           writeln ;
         end ;
         inc (ctrl) ;
         inc (p , o) ;
        end ;
end ;
 
begin
        assign (input , 'shuttle.in') ; reset (input) ;
        assign (output , 'shuttle.out') ; rewrite (output) ;
 
        readln (n) ;
        dfs (1) ;
        if (ctrl mod 20 <> 1) then
         writeln ;
 
        close (input) ; close (output) ;
end .

var n,tot:longint;
    a:array[0..13,0..200] of longint;

procedure dfs(x,shou,fang:longint);
var t,i:longint;
begin
 if x>n then exit;
 if fang=1 then t:=2 else t:=-2;
 a[x,1]:=shou;
 for i:=2 to x+1 do a[x,i]:=a[x,i-1]+t;
 if x=n then exit;
 if fang=1 then dfs(x+2,shou-2,1)
           else dfs(x+2,shou+2,2);
end;
//=============================================
procedure print;
var i,j:longint;
begin
 tot:=0;
 for i:=1 to n do
 for j:=1 to i+1 do
 begin
  if tot=20 then begin writeln; tot:=0; end;
  write(a[i,j]); inc(tot);
  if tot<>20 then write(' ');
 end;
 for i:=n-1 downto 1 do
 for j:=1 to i+1 do
 begin
  if tot=20 then begin writeln; tot:=0; end;
  write(a[i,j]); inc(tot);
  if tot<>20 then write(' ');
 end;
 if tot=20 then writeln;
 write(n+1);
 writeln;
end;
//=============================================

program shuttle;
var
  n,i,j,k,x,line:integer;
begin
  assign(input,'shuttle.in');reset(input);
  assign(output,'shuttle.out');rewrite(output);
  readln(n);
  for i:=1 to n+n
  do begin
     if i<=n then k:=i+1 else k:=n+n+1-i;
     if odd(i)
     then begin
          x:=n-k;
          for j:=1 to k
          do begin
             inc(x,2);
             inc(line);
             if i=n+n then writeln(x) else
             if line mod 20=0 then writeln(x) else write(x,' ');
             end;
          end
     else begin
          x:=n+k+2;
          for j:=1 to k
          do begin
             dec(x,2);
             inc(line);
             if i=n+n then writeln(x) else
             if line mod 20=0 then writeln(x) else write(x,' ');
             end;
          end;
     end;
  close(output);
end.

program cch(input,output);
type
 node=record
       s:string[25];
       from:longint;
       p:longint;
      end;
var
 n,c1:integer;
 a:array[1..150000] of node;
 head,tail:longint;
 
procedure init;
begin
 assign(input,'shuttle.in');
 assign(output,'shuttle.out');
 reset(input);
 rewrite(output);
 readln(n);
 c1:=0;
end;
 
procedure print(k:longint);
begin
 if k<>0 then
  begin
   print(a[k].from);
   if k<>tail then
    begin
     if k<>1 then
      begin
       inc(c1);
       if c1 mod 20=0 then writeln(a[k].p)
        else write(a[k].p,' ');
      end;
    end
   else writeln(a[k].p);
  end;
end;
 
procedure main;
var
 s0,s:string;
 i:longint;
begin
 s0:=''; s:='';
 for i:=1 to n do s:=s+'W';
 for i:=1 to n do s0:=s0+'B';
 s:=s+' '; s0:=s0+' ';
 for i:=1 to n do s:=s+'B';
 for i:=1 to n do s0:=s0+'W';
 head:=1; tail:=1;
 a[head].s:=s; a[head].from:=0; a[head].p:=n+1;
 repeat
  s:=a[head].s;
  if (a[head].p-1>0)and(s[a[head].p-1]='W') then
   begin
    inc(tail); a[tail].s:=s; a[tail].s[a[head].p]:='W';
    a[tail].s[a[head].p-1]:=' '; a[tail].p:=a[head].p-1; a[tail].from:=head;
    if a[tail].s=s0 then
     begin
      print(tail); break;
     end;
   end;
  if (a[head].p-2>0)and(s[a[head].p-2]='W')and(s[a[head].p-1]='B') then
   begin
    inc(tail); a[tail].s:=s;
    a[tail].s[a[head].p]:='W'; a[tail].s[a[head].p-2]:=' ';
    a[tail].p:=a[head].p-2; a[tail].from:=head;
    if a[tail].s=s0 then
     begin
      print(tail); break;
     end;
   end;
  if (a[head].p<=2*n)and(s[a[head].p+1]='B') then
   begin
    inc(tail); a[tail].s:=s; a[tail].s[a[head].p]:='B';
    a[tail].s[a[head].p+1]:=' ';
    a[tail].p:=a[head].p+1; a[tail].from:=head;
    if a[tail].s=s0 then
     begin
      print(tail); break;
     end;
   end;
  if (a[head].p+2<=2*n+1)and(s[a[head].p+2]='B')and(s[a[head].p+1]='W') then
   begin
    inc(tail); a[tail].s:=s; a[tail].s[a[head].p]:='B';
    a[tail].s[a[head].p+2]:=' ';
    a[tail].p:=a[head].p+2; a[tail].from:=head;
    if a[tail].s=s0 then
     begin
      print(tail); break;
     end;
   end;
  inc(head);
 until head>tail;
 close(input);
 close(output);
end;
 
begin
 init;
 main;
end.

program ex411;
  const
    maxnode=200000;
    maxstep=2000;
  var
      n:        longint;
      q:        array[1..maxnode] of record
                s,k:      longint;
                end;
      step,fa:     array[1..maxnode] of longint;
      ans:      array[0..maxstep] of longint;
 
 
  procedure Init;
 
    begin
      assign(input,'shuttle.in'); reset(input);
      readln(n);
    end;
 
  function result(s,k:longint):boolean;
 
    begin
      if (s= (1 shl (n*2)-1) - (1 shl n-1) ) and (k=n+1)
         then exit(true) else exit(false);
    end;
 
  procedure print(tail:longint);
 
    begin
      if fa[tail]=0 then exit;
      print(fa[tail]);
      inc(ans[0]); ans[ans[0]]:=q[tail].k;
    end;
 
 
  procedure Main;
    var
         s,k,head,tail:    longint;
         news,newk:     longint;
         sk1,sk2,sk:    longint;
 
    begin
      s :=1 shl n-1;   k:=n+1;
 
      head :=1; tail :=1;  q[1].s :=s; q[1].k:=k;
 
      repeat
        s:=q[head].s;  k :=q[head].k;
 
        if  k > 2   then
            begin  sk1 :=1 shl (k-2) and s;
                   sk2 :=1 shl (k-3) and s;
                   if (sk1 =0) and (sk2>0)
                      then begin
                           sk :=1 shl (k-2)+1 shl (k-3);
                           news  :=s xor sk;   newk :=k-2;
                                  inc(tail);
                                  q[tail].s:=news;  q[tail].k:=newk;
                                  step[tail]:=step[head]+1;
                                  fa[tail] :=head;
                                  if result(news,newk) then begin
                                     print(tail); exit;
                                     end;
                           end;
                   end;
 
        if k > 1   then begin
                   if   s and (1 shl (k-2)) >0 then begin
                   news :=s;   newk :=k-1;
                        inc(tail);
                        q[tail].s:=news;   q[tail].k :=newk;
                        step[tail] :=step[head]+1;
                        fa[tail] :=head;
                        if result(news,newk) then begin
                                     print(tail); exit;
                                     end;
                        end;
                   end;
 
 
        if k < n*2+1   then begin
                       if  s and (1 shl (k-1)) =0 then begin
                   news :=s;  newk :=k+1;
                        inc(tail);
                        q[tail].s:=news;   q[tail].k:=newk;
                        step[tail] :=step[head]+1;
                        fa[tail] :=head;
                        if result(news,newk) then begin
                                     print(tail); exit;
                                     end;
                        end;
                   end;
 
        if k <n*2  then begin
                   sk1 :=1 shl (k-1) and s;
                   sk2 :=1 shl (k) and s;
                   if  (sk1 >0) and (sk2=0)
                      then begin
                           sk :=1 shl (k-1)+1 shl (k);
                           news  :=s xor sk;   newk :=k+2;
                                  inc(tail);
                                  q[tail].s:=news;  q[tail].k:=newk;
                                  step[tail]:=step[head]+1;
                                  fa[tail] :=head;
                                  if result(news,newk) then begin
                                     print(tail); exit;
                                     end;
                           end;
                   end;
      inc(head);
      until  head > tail;
    end;
 
  procedure print;
    var
        i:      longint;
    begin
      assign(output,'shuttle.out');  rewrite(output);
      for i :=1 to ans[0]-1 do begin 
          if i mod 20=0 then writeln(ans[i])
                        else write(ans[i],' ');
          end;      
      writeln(ans[ans[0]]);
      close(input);  close(output);
    end;
 
 
  begin
    Init;
    Main;
    print;
  end.
