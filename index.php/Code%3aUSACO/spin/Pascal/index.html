        flag,fl:boolean;
        v:array[0..6,1..2] of longint;

begin
        assign(input,'spin.in'); reset(input);
        assign(output,'spin.out'); rewrite(output);
        for i:=1 to 5 do
        begin
           read(v[i,1],v[i,2]);
           for j:=1 to v[i,2] do
           begin
             read(a[i,j,1],a[i,j,2]);
             a[i,j,2]:=(a[i,j,1]+a[i,j,2])mod 360;
           end;
           end;
        for i:=0 to 359 do
        begin
           for j:=0 to 359 do
             begin
               fl:=true;
               for k:=1 to 5 do
               begin
                for l:=1 to v[k,2] do
                  if ((a[k,l,1]<=a[k,l,2])and(j>=a[k,l,1])and(j<=a[k,l,2]))or((a[k,l,1]>a[k,l,2])and(j<=a[k,l,2])and(j+360>=a[k,l,1])) then
                   flag:=true;
                if not flag then begin fl:=false; break end else
                flag:=false;
               end;
              if fl then
              begin writeln(i); close(input); close(output);halt;  end;
             end;
             for p:=1 to 5 do
               for q:=1 to v[p,2] do
               begin
                 a[p,q,1]:=(a[p,q,1]+v[p,1])mod 360;
                 a[p,q,2]:=(a[p,q,2]+v[p,1]) mod 360;
               end;
        end;
        writeln('none');
       close(input);
       close(output);
end.

program spin;
var
  i,j,k,s:word;
  wheel:array[1..5,1..5,1..2]of word;
  speed:array[1..5]of word;
  num:array[1..5]of word;
  found:boolean;
  hash:array[0..359]of byte;
begin
  assign(input,'spin.in');
  assign(output,'spin.out');
  reset(input);
  rewrite(output);
  for i:=1 to 5 do
  begin
    read(speed[i]);
    read(num[i]);
    for j:=1 to num[i] do
      read(wheel[i,j,1],wheel[i,j,2]);
    readln;
  end;
  found:=false;
  for i:=0 to 359 do
  begin
    fillchar(hash,sizeof(hash),0);
    for j:=1 to 5 do
      for k:=1 to num[j] do
        for s:=wheel[j,k,1] to wheel[j,k,1]+wheel[j,k,2] do
          inc(hash[s mod 360]);
    for s:=0 to 359 do
      if hash[s]=5 then
      begin
        writeln(i);
        found:=true;
        break;
      end;
    if found then
      break;
    for j:=1 to 5 do
      for k:=1 to num[j] do
        wheel[j,k,1]:=(wheel[j,k,1]+speed[j]) mod 360;
  end;
  if not found then
    writeln('none');
  close(output);
end.

program c050;
const
    inf = 'spin.in';
    ouf = 'spin.out';
 
type
    gap     = record
        b,e : integer;
    end;
 
    wheel   = record
        v   : byte;
        n   : byte;
        gs  : array[1..5] of gap;
    end;
 
var
    ws,now  : array[1..5] of wheel;
    time    : longint;
    com     : array[0..359] of integer;
    ans     : longint;
 
procedure readin;
var
    i,j:byte;
begin
    for i:= 1 to 5 do
        begin
            read(ws[i].v,ws[i].n);
            for j:= 1 to ws[i].n do
                begin
                    read(ws[i].gs[j].b,ws[i].gs[j].e);
                    inc(ws[i].gs[j].e,ws[i].gs[j].b);
                    ws[i].gs[j].b:=ws[i].gs[j].b mod 360;
                    ws[i].gs[j].e:=ws[i].gs[j].e mod 360;
                end;
            readln;
        end;
    now:=ws;
end;
 
function GoOn:boolean;
var
    i,j,k:integer;
begin
    fillchar(com,sizeof(com),0);
    for i:= 1 to 5 do
        begin
            for j:= 1 to now[i].n do
                begin
                    k:=now[i].gs[j].b;
                    while k<>now[i].gs[j].e do
                        begin
                            inc(com[k]);
                            inc(k);
                            if k=360 then
                                k:=0;
                        end;
                    inc(com[k]);
                end;
        end;
    GoOn:=true;
    for i:= 0 to 359 do
        if com[i]>=5 then
            exit(false);
end;
 
function Ended:boolean;
var
    i,j:byte;
begin
    ended:=true;
    for i:= 1 to 5 do
        begin
            for j:= 1 to ws[i].n do
                begin
                    if ws[i].gs[j].b<>now[i].gs[j].b then
                        exit(false);
                    if ws[i].gs[j].e<>now[i].gs[j].e then
                        exit(false);
                end;
        end;
end;
 
procedure next(var x:wheel);
var
    i:byte;
begin
    for i:= 1 to x.n do
        begin
            x.gs[i].b:=(x.gs[i].b + x.v) mod 360;
            x.gs[i].e:=(x.gs[i].e + x.v) mod 360;
        end;
end;
 
procedure finish;
begin
    writeln('none');
    close(input);
    close(output);
    halt;
end;
 
procedure work;
var
    i:byte;
begin
    while GoOn do
        begin
            inc(ans);
            for i:= 1 to 5 do
                next(now[i]);
            if Ended then
                finish;
        end;
end;
 
begin

var n,i,j,u,v,s:longint;o:boolean;
h,sd:array[1..5]of longint;
be,L,pasbe,nowbe:array[1..5,1..5]of longint;
dp:array[0..359]of longint;
 
function pd:boolean;
begin
  for i:=1 to 5 do
    for j:=1 to h[i] do
      if (be[i,j]<>nowbe[i,j]) then
        exit(false);
  exit(true);
end;
 
procedure add;
begin
         while v<>L[i,j] do
         begin
            u:=u mod 360;
            inc(dp[u],i);
            if dp[u]=15 then begin o:=true;break;end;
            inc(u);inc(v);
         end;
end;
 
procedure delete;
begin
  while v<>L[i,j] do
    begin
      u:=u mod 360;
      dec(dp[u],i);
      inc(u);inc(v);
    end;
end;
 
begin
  assign(input,'spin.in');reset(input);
  assign(output,'spin.out');rewrite(output);
  o:=false;
  for i:=1 to 5 do
    begin
      read(sd[i],h[i]);
      for j:=1 to h[i] do
        begin
          read(be[i,j],L[i,j]);
          pasbe[i,j]:=be[i,j];
          nowbe[i,j]:=(be[i,j]+sd[i]) mod 360;
          u:=be[i,j];v:=0;add;
        end;
    end;
  s:=0;
  if not o then
      repeat
        if pd then break;

            end;
        if o then break;
      until false;
 
  if o then writeln(s)
