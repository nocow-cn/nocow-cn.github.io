{

program fence4;
 
type
  TReal=double;//实数类型，也可以是 extended
  point=record
    x,y:Treal;
  end;//点类型
  line=record
    psta,pend:point;
  end;//有向线段类型
  vector=record
    vx,vy:Treal;
  end;//向量类型
 
const
  fn='fence4';
  zero=1e-4;
  maxp=200;
 
operator ** (p,q:vector) r:Treal;//计算叉积，用了平时不用的运算符“**”
begin
  r:=p.vx*q.vy-p.vy*q.vx;
end;
 
function LineToVec(l:line):vector;//有向线段转化成向量
begin
  LineToVec.vx:=l.pend.x-l.psta.x;
  LineToVec.vy:=l.pend.y-l.psta.y;
end;
 
function PointsToVec(ps,pe:point):vector;//起点和终点生成向量
begin
  PointsToVec.vx:=pe.x-ps.x;
  PointsToVec.vy:=pe.y-ps.y;
end;
 
function Cross(l1,l2:line):boolean;//判断线段相交，用跨立实验
begin
  if (LineToVec(l1)**PointsToVec(l1.psta,l2.psta))*(LineToVec(l1)**PointsToVec(l1.psta,l2.pend))>=0
  then exit(false);
  if (LineToVec(l2)**PointsToVec(l2.psta,l1.psta))*(LineToVec(l2)**PointsToVec(l2.psta,l1.pend))>=0
  then exit(false);
  exit(true);
end;//“>=”意味着不计端点
 
procedure WriteEdge(l:line);
begin
  writeln(l.psta.x:0:0,' ',l.psta.y:0:0,' ',l.pend.x:0:0,' ',l.pend.y:0:0);
end;
 
var
  sgt:point;
  n:longint;
  Edge:array [1..maxp] of line;
 
function Shade(l:line;p:point):boolean;//判断线段是否遮住点，也是跨立
begin
  if (PointsToVec(sgt,p)**PointsToVec(sgt,l.psta))*(PointsToVec(sgt,p)**PointsToVec(sgt,l.pend))>0
  then exit(false);
  if (LineToVec(l)**PointsToVec(l.psta,sgt))*(LineToVec(l)**PointsToVec(l.psta,p))>0
  then exit(false);
  exit(true);
end;//“>”意味着端点也算
 
procedure init;
var
  i:longint;
  tmp:line;
  Pts:array [1..maxp] of point;
begin
  readln(n);
  readln(sgt.x,sgt.y);
  fillchar(Pts,sizeof(Pts),0);
  fillchar(Edge,sizeof(Edge),0);
  for i:=1 to n do
    readln(Pts[i].x,Pts[i].y);
  for i:=1 to n-1 do
  begin
    Edge[i].psta:=Pts[i];
    Edge[i].pend:=Pts[i+1];
  end;
  Edge[n].psta:=Pts[1];
  Edge[n].pend:=Pts[n];
  tmp:=Edge[n];Edge[n]:=Edge[n-1];Edge[n-1]:=tmp;//这三行表示按题目要求处理边的顺序
end;
 
procedure main;
var
  i,j,cal:longint;
  concern,seen:array [1..maxp] of boolean;
 
function see(ps,pe:point):boolean;
var
  ij:longint;
  t1,t2,f1,f2:boolean;
  pm:point;
begin
  if (abs(pe.x-ps.x)<zero)and(abs(pe.y-ps.y)<zero) then exit(false);//线段太短的就当做看不见（必须的）
  f1:=true;f2:=true;//端点是否可见
  for ij:=1 to n do
    if (ij<>i)and concern[ij] then
      begin
        t1:=Shade(Edge[ij],ps);
        t2:=Shade(Edge[ij],pe);
        if t1 and t2 then exit(false);//若两端点均被同一线段遮挡，则不可见
        f1:=f1 and not t1;
        f2:=f2 and not t2;
      end;
  if f1 or f2 then exit(true);//若端点之一可见，则线段可见
  pm.x:=(ps.x+pe.x)/2;
  pm.y:=(ps.y+pe.y)/2;//线段中点
  if see(ps,pm) then exit(true);//二分递归
  see:=see(pm,pe);//二分递归
end;
 
begin
  cal:=0;
  for i:=1 to n do
    for j:=1 to n do
      if Cross(Edge[i],Edge[j]) then
        begin
          writeln('NOFENCE');
          exit;
        end;
  fillchar(concern,sizeof(concern),true);
  fillchar(seen,sizeof(seen),true);
  for i:=1 to n do
    concern[i]:=not ((Edge[i].psta.x-sgt.x)*(Edge[i].pend.y-Edge[i].psta.y)
                      =(Edge[i].psta.y-sgt.y)*(Edge[i].pend.x-Edge[i].psta.x));
  //如果原点和线段两端点共线，则不可见，也不会挡住其他线段
  for i:=1 to n do
  begin
    if concern[i] then seen[i]:=see(Edge[i].psta,Edge[i].pend)
    else seen[i]:=false;
    if seen[i] then inc(cal);
  end;
  writeln(cal);
  for i:=1 to n do
    if seen[i] then
      WriteEdge(Edge[i]);
end;
 
begin
  assign(input,fn+'.in');reset(input);
  assign(output,fn+'.out');rewrite(output);
  init;
  main;
  close(input);close(output);
end.

function cross(x1,y1,x2,y2:real; k:node):boolean;
var f:boolean;
    s1,s2:real;
begin
  f:=true;
  s1:=(x1-x2)*(y1-k.y1)-(y1-y2)*(x1-k.x1);
  s2:=(x1-x2)*(y1-k.y2)-(y1-y2)*(x1-k.x2);
  f:=f and (s1*s2<=0);
  s1:=(k.x1-k.x2)*(k.y1-y1)-(k.y1-k.y2)*(k.x1-x1);
  s2:=(k.x1-k.x2)*(k.y1-y2)-(k.y1-k.y2)*(k.x1-x2);
  f:=f and (s1*s2<=0);
  cross:=f;
end;
