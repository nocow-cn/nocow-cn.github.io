program holstein;
  type health=array[1..25]of integer;  //维生素数组,或维生素向量
  var ing:array[1..15]of health;
      req,nog:health;
      ans,nowa:array[1..15]of integer;
      v,g,i,j,totn,nown:integer;
  operator +(a,b:health)c:health;
    var i:integer;
    begin
      fillchar(c,sizeof(c),0);
      for i:=1 to v do
        c[i]:=a[i]+b[i];
    end;
  operator -(a,b:health)c:health;
    var i:integer;
    begin
      fillchar(c,sizeof(c),0);
      for i:=1 to v do
        c[i]:=a[i]-b[i];
    end;
  function check:boolean;  //营养是否足够
    var i:integer;
    begin
      for i:=1 to v do
        if nog[i]<req[i]
          then exit(false);
      exit(true);
    end;
  procedure update;  //记录最优解
    var i:integer;
    begin
      fillchar(ans,sizeof(ans),0);
      for i:=1 to nown do
        ans[i]:=nowa[i];
      totn:=nown;
    end;
  procedure dfs(now:integer);
    begin
      if nown>totn  //不可能最优
        then exit;
      if check  //得到解
        then begin
          update;
          exit;
        end;
      if now=g+1  //无解
        then exit;
      dfs(now+1);  //不取第now个饲料
       inc(nown);  //取第now个饲料
       nowa[nown]:=now;
      nog:=nog+ing[now];
      dfs(now+1);
      nog:=nog-ing[now];
      dec(nown);
    end;
  begin
    assign(input,'holstein.in');
    assign(output,'holstein.out');
    reset(input);
    rewrite(output);
    fillchar(ing,sizeof(ing),0);
    fillchar(req,sizeof(req),0);
    readln(v);
    for i:=1 to v do
      read(req[i]);
    readln(g);
    for i:=1 to g do
      for j:=1 to v do
        read(ing[i,j]);
    fillchar(ans,sizeof(ans),0);
    fillchar(nog,sizeof(nog),0);
    fillchar(nowa,sizeof(nowa),0);
    nown:=0;
    totn:=g;
    dfs(1);
    write(totn);
    for i:=1 to totn do
      write(' ',ans[i]);

var
  i,j,v,g,ans,best:longint;
  f:array[0..16] of boolean;
  vol,s2,s1:array[0..26] of longint;
  w:array[0..16,0..26] of longint;
 
 
function ok:boolean;
var
  i:longint;
  begin
    for i:=1 to v do
      if vol[i]>0 then exit(false);
    exit(true);
  end;
 
procedure update;
var
  i,j:longint;
  begin
    for i:=1 to ans do
      s2[i]:=s1[i];
    best:=ans;
  end;
 
procedure dfs(s,dep:longint);
var
  i,j:longint;
  begin
    if ok and (dep<best) then update;
    if dep+1<best then
    for i:=s+1 to g do
      begin
        if dep+1<best then begin
        for j:=1 to v do
          vol[j]:=vol[j]-w[i,j];
        inc(ans);
        s1[ans]:=i;
        dfs(i,dep+1);
        for j:=1 to v do
          vol[j]:=vol[j]+w[i,j];
        dec(ans);
        end;
      end;
  end;
