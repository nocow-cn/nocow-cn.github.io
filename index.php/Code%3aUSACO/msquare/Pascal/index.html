var q:array[1..40320] of state;
    visit:array[0..40319] of boolean;
    pre:array[1..40320] of longint;
    use:array[1..40320] of char;
    goal,tem:state;
    i,j,hash,h,f,e,m,step,s:longint;
    ans:string;
 
function calc(now:state):longint;
var ans,i,j,t:longint;
begin
  ans:=0;
  for i:=1 to 7 do
  begin
    t:=0;
    for j:=i+1 to 8 do if now[j]<now[i] then inc(t);
    ans:=ans*(9-i)+t;
  end;
  exit(ans);
end;
