void jia(long *a,long *b,long* c){
	for(long i=1;i<=max(a[0],b[0]);i++)
		c[i]=a[i]+b[i];
	c[0]=max(a[0],b[0]);
	for(long i=1;i<=c[0];i++){
		c[i+1]+=(c[i]/S);
		c[i]%=S;
	}
	if(c[c[0]+1])c[0]++;
}
void jian(long* a,long* b,long* c){
	for(long i=1;i<=max(a[0],b[0]);i++)
		c[i]=a[i]-b[i];
