begin

procedure swap(var i,j:node);
var temp:node;
begin temp:=i;i:=j;j:=temp;end;
procedure down(x:longint);
begin
  while x shl 1<=size do
  begin
    x:=x shl 1;
    if (x<size) and (h[x].time>h[x+1].time) then inc(x);
    if h[x].time<h[x shr 1].time then swap(h[x],h[x shr 1]) else break;
  end;
end;

begin
  for i:=1 to m1 do
  begin
    h[i].time:=a[i];h[i].id:=i;

  size:=m1;
  for i:=m1 shr 1 downto 1 do down(i);
  for i:=1 to n do
  begin
    aa[i]:=h[1].time;
    inc(h[1].time,a[h[1].id]);
    down(1);
  end;
  write(aa[n],' ');
end;
procedure work_b;
begin
  for i:=1 to m2 do

    h[i].time:=b[i];h[i].id:=i;

  begin
    bb[i]:=h[1].time;
    inc(h[1].time,b[h[1].id]);
    down(1);
  end;
  j:=0;
  for i:=1 to n do if aa[i]+bb[n-i+1]>j then j:=aa[i]+bb[n-i+1];
  writeln(j);
  close(input);close(output);
end;
begin
  init;

var costa,costb:array[1..1000] of longint;
    t,n,t1,t2,max,l,r,n1,n2,m,tmp,i,j,rest,k:longint;
    a,b:arr;
 
procedure qsort(l,r:longint; var a:arr);
var m,i,j,k:longint;
begin
 m:=a[(l+r) div 2];
 i:=l; j:=r;
 while i<j do
   begin
     while a[i]<m do inc(i);
     while a[j]>m do dec(j);
     if i<=j then begin
        k:=a[i]; a[i]:=a[j]; a[j]:=k;
        inc(i); dec(j);
        end;
   end;
 if i<r then qsort(i,r,a);
 if l<j then qsort(l,j,a);
end;
