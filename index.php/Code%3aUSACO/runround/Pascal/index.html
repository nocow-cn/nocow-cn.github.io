VAR
  a:array [1..10] of longint;
  i,j,k,n,vis,len,now,all,goal:longint;
  s:string;
  t:boolean;
BEGIN
  assign(input,'runround.in');
  assign(output,'runround.out');
  reset(input);
  rewrite(output);
  readln(s);
  val(s,n);
    repeat
      inc(n);
      str(n,s);
        if pos('0',s)<>0 then continue;
      len:=length(s);
      t:=true;
        for i:=1 to len-1 do
          for j:=i+1 to len do
            if s[i]=s[j] then
              BEGIN
                t:=false;
                break;
              END;
        if t=false then continue;
        for i:=1 to len do a[i]:=ord(s[i])-48;
      vis:=0;
      vis:=vis or (1 shl (len-1));
      k:=1;
      now:=1;
      goal:=1;
      all:=trunc(exp(len*ln(2)))-1;
        repeat
          k:=(now+a[now]-1) mod len+1;
          now:=k;
            if (now=goal) and (vis=all) then
              BEGIN
                writeln(n);
                close(input);
                close(output);
                halt;
              END;
            if (vis shr (len-k)) and 1=1 then break;
          vis:=vis or (1 shl (len-k));
        until false;
    until false;
END.
