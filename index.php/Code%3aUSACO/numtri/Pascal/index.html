var
    a,f:array[0..1001,0..1001] of longint;
    n,i,j,ans:longint;
function max(x,y:longint):longint; begin if x>y then exit(x) else exit(y); end;
begin
    assign(input,'numtri.in');reset(input);
    assign(output,'numtri.out');rewrite(output);
    readln(n);
    for i:=1 to n do begin
        for j:=1 to i do read(a[i,j]);
        readln;
    end;
    f[1,1]:=a[1,1];     //每个动态规划都有初值，只不过有些默认为0所以省略不写
    for i:=2 to n do begin
        for j:=1 to i do
            f[i,j]:=max(f[i-1,j-1],f[i-1,j])+a[i,j];     //转移方程，参考题解页
    end;
    ans:=0;
    for i:=1 to n do if ans<f[n,i] then ans:=f[n,i];      //求最大的答案
    writeln(ans);
    close(input);close(output);
end.  //By-LYP

program numtri(input,output);
  var
    a:array[1..1000,1..1000]of longint;
    sum:array[1..1000,0..1000]of longint;
    ans,r,i,j,k:longint;
  function max(i,j:longint):longint;
    begin
      if i>j
        then exit(i)
        else exit(j);
    end;
  begin
    assign(input,'numtri.in');reset(input);
    assign(output,'numtri.out');rewrite(output);
    read(r);
    for i:=1 to r do
      for j:=1 to i do
        read(a[i,j]);
    fillchar(sum,sizeof(sum),0);
    sum[1,1]:=a[1,1];
    for i:=2 to r do
      for j:=1 to i do
        sum[i,j]:=max(sum[i-1,j-1],sum[i-1,j])+a[i,j];
    ans:=0;
    for j:=1 to r do
      ans:=max(sum[r,j],ans);
    writeln(ans);
    close(input);
    close(output)
  end.
''  假设从顶至数字三角形的某一位置的所有路径中，
     所经过的数字的总和最大的那条路径，
     我们称之为该位置的最大路径。
     由于问题规定每一步只能沿直线向下或沿斜线向右下走，
     若要走到该位置，则其前一位置必为其左上方或正上方两个位置之一，
     由此可知，当前位置的最优路径必定与其左上方或正上方两个位置的最优路径有关，
     且来自于其中更优的一个。 
'''状态转移方程'''
sum[i，j]=max{sum[i-1，j]，sum[i-1，j-1]} + a[i，j] 
                  其中：2<=i<=n，2<=j<=i
'''边界条件：'''
sum[i，1]=sum[i-1，1]+a[i，1]     {第1列}
sum[i，i]=sum[i-1，i-1]+a[i，i]   {对角线}''

      f[i,j]:=f[i,j]+a[i,j];
   end;
   max:=0;
