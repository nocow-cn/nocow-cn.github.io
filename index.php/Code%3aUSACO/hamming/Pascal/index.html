procedure calham(numa,numb:integer);
var x:integer;
begin
    x:=numa xor numb;
    //作用：上一句得到两数二进制表示位数不同的个数，下面得到x二进制表示中1的个数
    x := (x and $55555555) + ((x shr 1) and $55555555);//参考了 http://www.matrix67.com/blog/archives/264
    x := (x and $33333333) + ((x shr 2) and $33333333);
    x := (x and $0F0F0F0F) + ((x shr 4) and $0F0F0F0F);
    x := (x and $00FF00FF) + ((x shr 8) and $00FF00FF);
    x := (x and $0000FFFF) + ((x shr 16) and $0000FFFF);
    if x>=d then
    begin
        graph[numa,numb]:=true;
        graph[numb,numa]:=true;
    end;
end;
