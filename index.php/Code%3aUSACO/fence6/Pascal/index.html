var
  le,sy,lss,rss:array[1..100]of integer;
  ls,rs:array[1..100,1..100]of integer;
  temp,gr:array[1..100,1..100]of longint;
  i,j,ii,jj,k,n:integer;
  r:longint;
 
procedure re;
begin
  assign(input,'fence6.in');reset(input);
  assign(output,'fence6.out');rewrite(output);
 
  readln(n);
 
  for i:=1 to n do
    for j:=1 to n do begin
      gr[i,j]:=1000000;
    end;
 
  for i:=1 to n do begin
    read(j,le[i],lss[i],rss[i]);
    sy[j]:=i;
    for j:=1 to lss[i] do begin
      read(ls[i,j]);
    end;
    for j:=1 to rss[i] do begin
      read(rs[i,j]);
    end;
  end;
 
  for i:=1 to n do begin
    for j:=1 to lss[i] do begin
      ls[i,j]:=sy[ls[i,j]];
      gr[i,ls[i,j]]:=le[i]+le[ls[i,j]];
    end;
    for j:=1 to rss[i] do begin
      rs[i,j]:=sy[rs[i,j]];
      gr[i,rs[i,j]]:=le[i]+le[rs[i,j]];
    end;
  end;
 
  temp:=gr;
 
  r:=maxlongint;
end;
 
function min(x,y:longint):longint;
begin
  if x<y then exit(x);
  exit(y);
end;
 
procedure work;
begin
  for k:=1 to n do begin
    for i:=1 to lss[k]do begin
      ii:=ls[k,i];
      if ii<k then
        for j:=1 to rss[k]do begin
          jj:=rs[k,j];
          if jj<k then
            r:=min(r,gr[ii,jj]+temp[k,ii]+temp[k,jj]);
        end;
    end;
    for i:=1 to n do  for j:=1 to n do  gr[i,j]:=min(gr[i,j],gr[i,k]+gr[k,j]);
  end;
end;
 
begin
  re;
 
  work;
 
  writeln(r div 2);
  close(output);
end.

    visit:array[1..100] of boolean;
    edge:array[1..100] of integer;

    c:array[1..100,1..2,0..8] of integer;
procedure init;
var i,j,x:integer;
begin
  assign(input,'fence6.in');
  reset(input);
  readln(n);
  for i:=1 to n do
  begin read(x);
        readln(edge[x],c[x,1,0],c[x,2,0]);
        for j:=1 to c[x,1,0] do
        begin read(c[x,1,j]);
              left[x,c[x,1,j]]:=true;
        end;
        readln;
        for j:=1 to c[x,2,0] do
                 read(c[x,2,j]);
        readln;
   end;
   close(input);
end;
 
procedure DFS(v,gfr,dis:integer);
var sfr,now,i,gto:integer;
begin
  if dis>=min then exit;
  if gfr=1 then gto:=2 else gto:=1;
  for i:=1 to c[v,gto,0] do
  begin now:=c[v,gto,i];
        if left[now,v] then sfr:=1 else sfr:=2;
        if (now=start)and(sfr=1) then
          if dis<=min then
          begin
             min:=dis;
             exit;
          end;
        if not visit[now] then
        begin visit[now]:=true;
              DFS(now,sfr,edge[now]+dis);
              visit[now]:=false;
        end;
   end;
end;

begin

  end;
  assign(output,'fence6.out');
  rewrite(output);
  writeln(min);

var
	n,k,i,j,n1,n2,s,x,node,ans:integer;
	len:array[1..100]of integer;
	g:array[1..100,1..100]of integer;
	f:array[1..200,1..200]of integer;
        b:array[1..200]of boolean;
        dist:array[1..200]of integer;
procedure dijkstra(x:integer);
        var
                i,j,w,min:integer;
        begin
                fillchar(b,sizeof(b),0);
                fillword(dist,sizeof(dist)div 2,maxint);
                dist[2*x-1]:=0;
                for i:=1 to 2*node-1 do begin
                        min:=maxint;
                        for j:=1 to 2*node do
                                if not b[j] and (dist[j]<min)then begin
                                        w:=j;min:=dist[j];
                                end;
                        b[w]:=true;
                        for j:=1 to 2*node do
                                if not b[j] and(dist[w]+f[w,j]<dist[j])then
                                        dist[j]:=dist[w]+f[w,j];
                end;
        end;
begin
	assign(input,'fence6.in');reset(input);
	readln(n);node:=0;
	fillword(len,sizeof(len)div 2,maxint);
	fillchar(g,sizeof(g),$FF);
	for i:=1 to n do begin
		readln(s,x,n1,n2);
		len[s]:=x;
		if s>node then node:=s;
		for j:=1 to n1 do begin
			read(x);
			g[s,x]:=0;
		end;
		readln;
		for j:=1 to n2 do begin
			read(x);
			g[s,x]:=1;
		end;
		readln;
	end;
	close(input);
	fillword(f,sizeof(f) div 2,maxint);
	for i:=1 to node do begin
		f[2*i-1,2*i]:=len[i];
		f[2*i,2*i-1]:=len[i];
	end;
	for i:=1 to node do
		for j:=1 to node do
			if g[i,j]<>-1 then f[2*i-g[i,j],2*j-g[j,i]]:=0;
        ans:=maxint;
        for i:=1 to node do begin
                f[2*i-1,2*i]:=maxint;
                dijkstra(i);
                f[2*i-1,2*i]:=len[i];
                if dist[2*i]+len[i]<ans then ans:=dist[2*i]+len[i];
        end;
	assign(output,'fence6.out');rewrite(output);
	writeln(ans);
	close(output);
end.
