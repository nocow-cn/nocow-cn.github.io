var
    a:array[0..201,1..2] of longint;
    b:array[0..200] of boolean;
    n,m,c,i,j,t,ans,k:longint;
    p:boolean;
begin
    assign(input,'barn1.in');reset(input);
    assign(output,'barn1.out');rewrite(output);
    readln(n,t,c);
    ans:=c;
    for i:=1 to c do begin readln(a[i,1]); a[i,2]:=a[i,1]; end;
    for i:=1 to c-1 do    //简单排序
        for j:=i+1 to c do
            if a[i,1]>a[j,1] then begin
                a[0]:=a[i];
                a[i]:=a[j];
                a[j]:=a[0];
            end;
    fillchar(b,sizeof(b),0);
    for i:=2 to c do
        b[a[i,1]-a[i-1,2]]:=true;    //标记空隙的大小
    i:=0;
    a[201,1]:=0;    //这个......是用来后面合并木板时用的
    a[201,2]:=0;
    while ans>n do begin
        k:=2;
        repeat inc(i); until b[i];     //找一个空隙小的
        while (k<=c) and (ans>n) do begin
            p:=true;
            if a[k,1]-a[k-1,2]=i then begin    //合并
                p:=false;
                a[k-1,2]:=a[k,2];
                for j:=k to c do a[j]:=a[j+1];      //删除多余的木板，刚刚两条语句......
                dec(c);
                dec(ans);
            end;
            if p then inc(k);
        end;
    end;
    t:=0;
    for i:=1 to c do inc(t,a[i,2]-a[i,1]+1);    //求答案
    writeln(t);
    close(input);close(output);
end. //By-LYP

procedure qsort(l,r:longint);
var i,j,x,t:longint;
begin
 i:=l; j:=r;
 x:=a[l+random(r-l)];
 repeat
  while a[i]>x do inc(i);
  while a[j]<x do dec(j);
  if i<=j then
  begin
   t:=a[i]; a[i]:=a[j]; a[j]:=t;
   inc(i); dec(j);
  end;
 until i>j;
 if i<r then qsort(i,r);
 if l<j then qsort(l,j);
end;

var i,min,max:longint;
begin
 readln(tot,n,n);
 min:=maxlongint; max:=-maxlongint;

 ans:=max-min;

var

  occ:array[1..200]of boolean;
  heap:array[1..200]of byte;

procedure add (len:byte);
var
  i:byte;
begin
  heap[em]:=len;
  i:=em;
  while (i<>1) and (heap[i div 2]<len) do
  begin
    heap[i]:=heap[i div 2];
    heap[i div 2]:=len;
    i:=i div 2;
  end;
  while heap[em]<>0 do
    inc(em);
end;
//
procedure fix (s:byte);
var
  i,j:byte;
begin
  i:=s*2;
  j:=s*2+1;
  if (i>200) or (heap[i]=0) and (heap[j]=0) then
  begin
    heap[s]:=0;
    if s<em then
      em:=s;
    exit;
  end;
  if heap[i]<heap[j] then
  begin
    heap[s]:=heap[j];
    fix(j);
  end
  else
  begin
    heap[s]:=heap[i];
    fix(i);
  end;
end;

    dec(e);

    if not occ[i] then
    begin
      j:=i+1;

      i:=j+1;
    end
    else
      inc(i);

  begin
    dec(tot,heap[1]);

const
  maxc=205;
  maxm=55;
var
  m,s,c,i,j,k,min:longint;
  a:array[1..maxc]of longint;
  f:array[1..maxc,0..maxm]of longint;   {前i头牛用j个隔板的最小长度}
procedure qsort(l,r:longint);
var i,j,x,tmp:longint;
begin
  i:=l; j:=r; x:=a[(l+r) shr 1];
  repeat
    while a[i]<x do inc(i);
    while a[j]>x do dec(j);
    if i<=j then
    begin
      tmp:=a[i]; a[i]:=a[j]; a[j]:=tmp;
      inc(i); dec(j);
    end;
  until i>j;
  if l<j then qsort(l,j);
  if i<r then qsort(i,r);
end;
begin
  assign(input,'barn1.in'); reset(input);
  assign(output,'barn1.out'); rewrite(output);
  readln(m,s,c);
  for i:=1 to c do readln(a[i]);
  qsort(1,c);
  for i:=1 to c do
    f[i,1]:=a[i]-a[1]+1;
  if m>c then m:=c;
  for i:=1 to c do
    for j:=2 to m do
    begin
      min:=maxlongint;
      for k:=j-1 to i-1 do
        if f[k,j-1]+a[i]-a[k+1]+1<min then min:=f[k,j-1]+a[i]-a[k+1]+1;
      f[i,j]:=min;
    end;
  writeln(f[c,m]);
  close(input); close(output);
end.

Program barn1;
 
Const
  fin                   =       'barn1.in';
  fout                  =       'barn1.out';
 
Type
  TypeArray             =       Array [0..201] of Longint;
 
Var
  m,s,c                 :       Longint;
  w,h                   :       TypeArray;
 
Procedure Init;
  var
    i                   :       Longint;
  begin
    readln(m,s,c);
    for i:=1 to c do readln(w[i]);
  end;
 
Procedure Qsort(l,r:Longint;var s:TypeArray);
  var
    i,j,t,mid           :       Longint;
  begin
    i:=l;j:=r;mid:=s[(l+r) shr 1];
    Repeat
      while s[i]>mid do inc(i);
      while s[j]<mid do dec(j);
      if i<=j
        then
          begin
            t:=s[i];s[i]:=s[j];s[j]:=t;
            inc(i);dec(j);
          end;
    Until i>=j;
    if l<j then Qsort(l,j,s);
    if i<r then Qsort(i,r,s);
  end;
 
Procedure Main;
  var
    i                   :       Longint;
  begin
    Qsort(1,c,w);
    for i:=1 to c-1 do h[i]:=w[i]-w[i+1]-1;
    Qsort(1,c-1,h);
  end;
 
Procedure Print;
  var
    i,Last              :       Longint;
  begin
    if m>=c then begin writeln(c);exit;end;
    Last:=w[1]-w[c]+1;
    for i:=1 to m-1 do Last:=Last-h[i];
    writeln(Last);
  end;
 
Begin
  Assign(INPUT,fin);Reset(INPUT);
  Init;
  Close(INPUT);
  Main;
  Assign(OUTPUT,fout);Rewrite(OUTPUT);
  Print;
  Close(OUTPUT);
End.
