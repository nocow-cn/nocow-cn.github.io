#include <stdio.h>
#include <math.h>
 
typedef struct{double x,y;}point;
 
int n;
point z[10001];
 
void first(){
	int i;
	scanf("%d",&n);
	for(i=1;i<=n;i++)
		scanf("%lf%lf",&z[i].x,&z[i].y);
}
 
int dblcmp(double x){
	if(fabs(x)<0.00001)return 0;
	else return x>0?1:-1;
}
 
double det(double x1,double y1,double x2,double y2){
	return x1*y2-x2*y1;
}
 
double cross(point a,point b,point c){
	return det(c.x-a.x,c.y-a.y,b.x-a.x,b.y-a.y);
}
 
int sn;
int stack[10001];
 
void swap(int i,int j){
	point k;
	k=z[i];z[i]=z[j];z[j]=k;
}
 
void P(int p,int r){
	if(p>=r)return;
	int i=p,j=r+1;
	swap(p,(p+r)/2);
	while(1){
		while((z[++i].y<z[p].y ||(z[i].y==z[p].y && z[i].x<z[p].x)) && i<=r);
		while((z[--j].y>z[p].y ||(z[j].y==z[p].y && z[j].x>z[p].x)) && j>=p);
		if(i>=j)break;
		swap(i,j);
	}
	swap(p,j);
	P(p,j-1);
	P(j+1,r);
}
 
double ans;
 
double dis(point a,point b){
	return sqrt((a.x-b.x)*(a.x-b.x)+(a.y-b.y)*(a.y-b.y));
}
 
void gramham(){
	int i;
	P(1,n);
	stack[0]=1;
	sn=1;
	for(i=2;i<=n;i++){
		while(sn>1){
			if(dblcmp(cross(z[stack[sn-2]],z[stack[sn-1]],z[i]))<0){
				sn--;
				stack[sn]=0;
			}else break;
		}
		stack[sn]=i;sn++;
	}
	for(i=n-1;i>=1;i--){
		while(sn>1){
			if(dblcmp(cross(z[stack[sn-2]],z[stack[sn-1]],z[i]))<0){
				sn--;
				stack[sn]=0;
			}else break;
		}
		stack[sn]=i;sn++;
	}
	for(i=0;i<sn-1;i++)
		ans+=dis(z[stack[i]],z[stack[i+1]]);
	printf("%.2lf\n",ans);
}
 
main(){
	freopen("fc.in","r",stdin);
	freopen("fc.out","w",stdout);
	first();
	gramham();
	return 0;
}

#include <stdio.h>
#include <math.h>
const phi=1e-7;
FILE *in,*out;
int alp[10001]={};
struct point {double x,y;}a[10001]={};
int dblcmp(double d)
{
  if (fabs(d)<phi) return 0;else return (d>0)?1:-1;
}
int dodet(struct point a,struct point b)
{
  return dblcmp(a.x*b.y-b.x*a.y);
}
int det(struct point a,struct point b,struct point c)
{
  struct point b1,c1;
  b1.x=b.x-a.x;
  b1.y=b.y-a.y;
  c1.x=c.x-a.x;
  c1.y=c.y-a.y;
  return dodet(b1,c1);
}
double dis(struct point a,struct point b)
{
  return sqrt((a.x-b.x)*(a.x-b.x)+(a.y-b.y)*(a.y-b.y));
}
void qsort(int l,int r)
{
  int i,j,t,x;
  i=l;j=r;x=alp[(l+r)/2];
  do
  {
    while ((a[alp[i]].y<a[x].y)||((a[alp[i]].y==a[x].y)&&(a[alp[i]].x<a[x].x))) i++;
    while ((a[alp[j]].y>a[x].y)||((a[alp[j]].y==a[x].y)&&(a[alp[j]].x>a[x].x))) j--;
    if (i<=j)
      {
	t=alp[i];
	alp[i++]=alp[j];
	alp[j--]=t;
      }
  }while (i<=j);
  if (l<j) qsort(l,j);
  if (i<r) qsort(i,r);
}
int main()
{
  double ans;
  int i,j,top,n,stk[10001]={};
  in=fopen("fc.in","r");
  fscanf(in,"%d\n",&n);
  for (i=1;i<=n;i++) 
    {
      fscanf(in,"%lf %lf\n",&a[i].x,&a[i].y);
      alp[i]=i;
    }
  fclose(in);  
  qsort(1,n);
  top=0;
  for (i=1;i<=n;i++)
    {
      while ((top>1)&&(det(a[stk[top-1]],a[stk[top]],a[alp[i]])<0)) top--;
      stk[++top]=alp[i];
    }
  j=top;
  for (i=n;i>=1;i--)
    {
      while ((top>j)&&(det(a[stk[top-1]],a[stk[top]],a[alp[i]])<0)) top--;
      stk[++top]=alp[i];
    }
  ans=dis(a[stk[1]],a[stk[top]]);
  for (i=2;i<=top;i++) ans+=dis(a[stk[i]],a[stk[i-1]]);
  out=fopen("fc.out","w");
  fprintf(out,"%.2lf\n",ans);
  fclose(out);
  return 0;
}
