#include <stdio.h>
#define MAX 200000000
 
long n,m;
long lnm[33][33];
long lin[1001][3];
long map[33][33];
long cpy[33][33];
long nam[33];
 
void first(){
	long i,j,k;
	scanf("%ld%ld",&n,&m);
	for(i=1;i<=m;i++){
		scanf("%ld%ld%d",&lin[i][0],&lin[i][1],&k);
		lin[i][2]=k;
		map[lin[i][0]][lin[i][1]]+=k;
		lnm[lin[i][0]][lin[i][1]]++;
	}
	for(i=1;i<=n;i++)
		for(j=1;j<=n;j++)
			cpy[i][j]=map[i][j];
}
 
long s,e;
long stack[33];
 
void bfs(){
	long i;
	s=0;e=1;
	stack[0]=n;
	for(i=1;i<=n;i++)nam[i]=MAX;
	nam[n]=0;
	while(s!=e){
		for(i=1;i<=n;i++)
			if(nam[i]==MAX && map[i][stack[s]]!=0){
				nam[i]=nam[stack[s]]+1;stack[e]=i;e++;
			}
		s++;
	}
}
 
long pre[33];
long num[33];
long ans;
 
void add(){
	long i=n,j=MAX;
	while(i!=1){
		if(j>map[pre[i]][i])j=map[pre[i]][i];
		i=pre[i];
	}
	ans+=j;i=n;
	while(i!=1){
		map[pre[i]][i]-=j;
		map[i][pre[i]]+=j;
		i=pre[i];
	}
}
 
void sap(){
	long i,j,k;
	for(i=1;i<=n;i++)
		for(j=1;j<=n;j++)
			map[i][j]=cpy[i][j];
	bfs();
	ans=0;
	i=1;
	while(nam[1]<n){
		k=MAX;
		if(i==n){add();i=1;}
		for(j=1;j<=n;j++)
			if(map[i][j]!=0 && nam[i]==nam[j]+1){
				pre[j]=i;
				i=j;
				break;
			}else if(map[i][j]!=0 && k>nam[j])k=nam[j];
		if(j==n+1){
			num[nam[i]]--;
			if(k!=MAX){
				nam[i]=nam[j]+1;
				nam[i]=k+1;
			}else nam[i]=MAX;
			if(i!=1)i=pre[i];
		}
	}
}
 
int isi[33][33];
int v[1001][2];
int is[1001];
 
void swap(long i,long j){
	long k;
	k=v[i][0];v[i][0]=v[j][0];v[j][0]=k;
	k=v[i][1];v[i][1]=v[j][1];v[j][1]=k;
}
 
void P(long p,long r){
	if(p>=r)return;
	long i=p,j=r+1;
	swap(p,(p+r)/2);
	while(1){
		while((v[++i][1]>v[p][1] || (v[i][1]==v[p][1] && v[i][0]<v[p][0]))&& i<=r);
		while((v[--j][1]<v[p][1] || (v[j][1]==v[p][1] && v[j][0]>v[p][0]))&& j>=p);
		if(i>=j)break;
		swap(i,j);
	}
	swap(p,j);
	P(p,j-1);
	P(j+1,r);
}
 
void doing(){
	long i,j,k=0,l,g=ans,f=0;
	for(i=1;i<=n;i++)
		for(j=1;j<=n;j++)
			if(map[i][j]==0 && cpy[i][j]!=0)isi[i][j]=1;
	for(i=1;i<=m;i++)
		if(isi[lin[i][0]][lin[i][1]]==1){v[k][0]=i;v[k][1]=lin[i][2];k++;}
	P(0,k-1);
	for(i=0;i<k;i++){
		l=cpy[lin[v[i][0]][0]][lin[v[i][0]][1]];
		cpy[lin[v[i][0]][0]][lin[v[i][0]][1]]=0;
		sap();
		if(g-ans==l){
			lin[v[i][0]][0]=-1;g=ans;f++;
		}
		else cpy[lin[v[i][0]][0]][lin[v[i][0]][1]]=l;
	}
	printf(" %ld\n",f);
	for(i=1;i<=m;i++)if(lin[i][0]==-1)printf("%ld\n",i);
}

#include <stdio.h>
#include <limits.h>
FILE *in,*out;
int n,m,ans[1001]={};
unsigned long f[33][33]={},c[33][33]={},rf;
struct redge
{
  long s,e,bh;
  unsigned long c;
}edge[1001]={},tmp;
unsigned long isap()
{
  int i,j,k,min,flag;
  unsigned long aug,flow=0,gap[1000]={},di[33],d[33]={},pv[33]={},pva[33]={};
  for (i=1;i<=n;i++) di[i]=1;
  aug=ULONG_MAX;
  i=1;
  pv[1]=1;
  gap[0]=n;
  while (d[n]<n) 
    {
      flag=0;
      pva[i]=aug;
      for (j=di[i];j<=n;j++)
	if ((f[i][j]>0)&&(d[j]+1==d[i]))
	  {
	    flag=1;
	    di[i]=j;
	    pv[j]=i;
	    if (f[i][j]<aug) aug=f[i][j];
	    i=j;
	    if (i==n)
	      {
		while (i!=1)
		  {
		    f[pv[i]][i]-=aug;
		    f[i][pv[i]]+=aug;
		    i=pv[i];
		  }
		flow+=aug;
		aug=ULONG_MAX;
	      }
	    break;
	  }
      if (!flag)
	{
	  min=n;
	  for (j=1;j<=n;j++)
	      if ((f[i][j]>0)&&(d[j]+1<min)) 
		{
		  min=d[j]+1;
		  k=j;
		}
	  di[i]=k;
	  if (!--gap[d[i]]) break;
	  gap[d[i]=min]++;
	  if (i!=1) aug=pva[i=pv[i]];
	}
    }
  return flow;
}

	{
	  tmp=edge[i];
	  edge[i]=edge[j];
	  edge[j]=tmp;
	}

  rf=isap();
  k=0;u=rf;

    {
      j=edge[i].c;

      f[edge[i].s][edge[i].e]-=j;

	}

    }
  for (i=1;i<=k-1;i++) for (j=i+1;j<=k;j++) if (ans[i]>ans[j]) {t=ans[i];ans[i]=ans[j];ans[j]=t;}
  out=fopen("milk6.out","w");
