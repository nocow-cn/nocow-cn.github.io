int main()
{
    ifstream fin("milk2.in");
    ofstream fout("milk2.out");
 
    int n, b, e, beg = 1000000, end = 0;
    pair<int, int> time_segment;
    vector< pair<int, int> > times;
 
    fin >> n;
    while (fin >> b >> e) {
        time_segment = make_pair(b, e);  // 把挤奶时间作为一个个时间段储存起来
        times.push_back(time_segment);
        // 找出挤奶的开始和结束时间
        if (beg > b) 
            beg = b;
        if (end < e)
            end = e;
    }
 
    int get_milk = 0, none_milk = 0, max_get = 0, max_none = 0;
    vector< pair<int, int> >::iterator iter;
    for (int t = beg; t <= end; ++t) {
 
        // 以时间为单位循环,每个时间点都把所有时间段搜索一遍
        for (iter = times.begin(); iter != times.end(); ++iter) { 
            // 时间点在某时间段内,则这一点有人挤奶
            if (iter->first <= t && t < iter->second) {
                ++get_milk;
                if (none_milk > max_none)
                    max_none = none_milk;
                none_milk = 0;
                break;
            }
        }
        // 搜索完所有时间段都没有匹配,则这一点无人挤奶
        if (iter == times.end()) {
            ++none_milk;
            if (get_milk > max_get)
                max_get = get_milk;
            get_milk = 0;
        }
    }
 
    fout << max_get << " " << max_none << endl;
    return 0;
}

void quicksort(long int a[], long int b[], int l, int r){
     int i=l-1,j=r,p=l-1,q=r;
     long int v=a[r];
     int k;
     if(r<=l) return;

        while(a[++i]<v);
        while(v<a[--j]) if(j==l) break;
        if(i>=j) break;
        exch(a[i],a[j]);
        exch(b[i],b[j]);
        if(a[i] == v){
          p++;
          exch(a[p],a[i]);
          exch(b[p],b[i]);
        }
        if(v==a[j]){
          q--;
          exch(a[j],a[q]);
          exch(b[j],b[q]);
        }
      }
      exch(a[i],a[r]);
      exch(b[i],b[r]);
      j=i-1; i=i + 1;
      for(k=l; k < p; k++, j--){
         exch(a[k], a[j]);
         exch(b[k], b[j]);
      }
      for(k=r-1;k>q;k--,i++){
         exch(a[i],a[k]);
         exch(b[i],b[k]);
      }
      quicksort(a, b, l, j);
      quicksort(a, b, i, r);
}

int partition(double A[],int begin,int end)
{
	int r = end;
	int i = begin - 1,j = begin;
	double temp;
 
	while(j < end)
	{
		if (A[j] < A[r])
		{
			i++;
			if(i != j)
			{
				temp = A[i];
				A[i] = A[j];
				A[j] = temp;
			}
		}
		j++;
	}
 
	if (i+1 != r)
	{
		temp = A[i+1];
		A[i+1] = A[r];
		A[r] = temp;
	}
 
	return i+1;
}
void Qsort(double A[],int begin,int end)
{
	if (begin < end)
	{
		int q = partition(A,begin,end);
		Qsort(A,begin,q-1);
		Qsort(A,q+1,end);
	}
}

typedef struct Farmertime
{
        int tm; bool isbegin ;
} farmertime;
 
bool myfunc( farmertime t1 ,farmertime t2) {
     bool b = false;
     if (t1.tm < t2.tm ) b=true ;
     else if (t1.tm == t2.tm) {
               if (t2.isbegin == false && t1.isbegin == true) b=true ;
          }
     return b ;
      }
