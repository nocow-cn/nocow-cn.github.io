program USACO_stamps;
 
const
  fn='stamps';
 
var
  n,k,maxi:word;
  a:array[0..50] of word;
  rec:array[0..2000000] of word;
 
procedure qsort(l,r:word);//快排，优化用
var
  i,j,x,t:word;
begin
  i:=l;j:=r;x:=a[(l+r)div 2];
  repeat
    while a[i]<x do inc(i);
    while a[j]>x do dec(j);
    if i<=j then
      begin
        t:=a[i];a[i]:=a[j];a[j]:=t;
        inc(i);dec(j);
      end;
  until i>=j;
  if i<r then qsort(i,r);
  if l<j then qsort(l,j);
end;
 
procedure init;
var
  i:word;
begin
  readln(k,n);maxi:=0;
  for i:=1 to n do
    begin
      read(a[i]);
      if a[i]>maxi then maxi:=a[i];
    end;
  qsort(1,n);
  fillword(rec,maxi*k+2,65535);
  rec[0]:=0;a[0]:=0;
end;
 
procedure main;
var
  i,j:longint;
begin
  for i:=1 to n do
    if a[i]>a[i-1] then  //似乎可有可无
      for j:=a[i] to a[i]*k do  //这里和快排一起构成了优化
        if rec[j]>rec[j-a[i]]+1 then rec[j]:=rec[j-a[i]]+1;//类似“松弛”
  i:=1;
  while rec[i+1]<=k do inc(i);
  writeln(i);
end;

procedure qsort(a,b:longint);
var l,r:longint;
mid,temp:longint;
begin
    if a=b then exit;
    l:=a;
    r:=b;
    mid:=data[(l+r)div 2];

        while data[l]<mid do inc(l);

        if l<=r then
        begin
            temp:=data[l];
            data[l]:=data[r];
            data[r]:=temp;
            inc(l);
            dec(r);
        end;
    until l>r;
    if a<r then qsort(a,r);
    if l<b then qsort(l,b);
end;
