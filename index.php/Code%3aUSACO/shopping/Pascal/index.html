program shopping;
const inf='shopping.in';
      ouf='shopping.out';
var
   m,ii,j,n,h:longint;
   c,k:array[1..99,1..5]of longint;
   hp:array[1..99]of longint;
   d,l,p,i,a:array[0..5]of longint;
   trans:array[0..999]of longint;
   f:array[0..5,0..5,0..5,0..5,0..5]of longint;
BEGIN
  assign(input,inf);reset(input);
  assign(output,ouf);rewrite(output);
  readln(m);
  for ii:=1 to m do
  begin
    read(h);
    for j:=1 to h do
      read(c[ii,j],k[ii,j]);
    readln(hp[ii])
  end;
  readln(n);
  for ii:=1 to n do
  begin
    readln(d[ii],l[ii],p[ii]);
    trans[d[ii]]:=ii;{将物品编号转化为1~5}
  end;
  i[1]:=-1;
  repeat{枚举每种物品}
   inc(i[1]);
   i[2]:=-1;
   repeat
    inc(i[2]);
    i[3]:=-1;
    repeat
     inc(i[3]);
     i[4]:=-1;
     repeat
      inc(i[4]);
      i[5]:=-1;
      repeat
       inc(i[5]);
       f[i[1],i[2],i[3],i[4],i[5]]:=i[1]*p[1]+i[2]*p[2]+i[3]*p[3]+i[4]*p[4]+i[5]*p[5];{初值}
       for j:=1 to m do{枚举每种优惠方案}
       begin
           a[1]:=i[1];a[2]:=i[2];a[3]:=i[3];a[4]:=i[4];a[5]:=i[5];
           a[trans[c[j,1]]]:=i[trans[c[j,1]]]-k[j,1];

           a[trans[c[j,4]]]:=i[trans[c[j,4]]]-k[j,4];
           a[trans[c[j,5]]]:=i[trans[c[j,5]]]-k[j,5];

  until i[1]=l[1];

  close(input);close(output);
