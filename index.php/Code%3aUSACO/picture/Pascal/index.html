	n := n shl 1;
end;
 
Procedure qsort(x, y, u:longint);
var
	left, right, t, g1, g2:longint;
begin
	left := x; right := y;
	g1 := h[(left + right) shr 1][u];
	g2 := flag[(left + right) shr 1][u];
	repeat
		while (h[left][u] < g1) or ((h[left][u] = g1) and (flag[left][u] > g2)) do inc(left);
		while (h[right][u] > g1) or ((h[right][u] = g1) and (flag[right][u] < g2)) do dec(right);
		if left <= right then
		begin
			t := h[left][u]; h[left][u] := h[right][u]; h[right][u] := t;
			t := l[left][u]; l[left][u] := l[right][u]; l[right][u] := t;
			t := r[left][u]; r[left][u] := r[right][u]; r[right][u] := t;
			t := flag[left][u]; flag[left][u] := flag[right][u]; flag[right][u] := t;
			inc(left); dec(right);
		end;
	until left > right;
	if x < right then qsort(x, right, u);
	if left < y then qsort(left, y, u);
end;
 
Procedure Build_tree(u:longint);
begin
	tree[u].sum := 0;
	tree[u].cover := true;
	if tree[u].p + 1 = tree[u].r then
		exit;
	tree[u].m := (tree[u].p + tree[u].r) shr 1;
	tree[u shl 1].p := tree[u].p;
	tree[u shl 1].r := tree[u].m;
	Build_tree(u shl 1);
	tree[u shl 1 + 1].p := tree[u].m;
	tree[u shl 1 + 1].r := tree[u].r;
	Build_tree(u shl 1 + 1);
end;
 
Procedure Change(u, a, b, v:longint);
begin
	if (tree[u].p = a) and (tree[u].r = b) then
	begin
		if tree[u].cover then
		begin
			inc(tree[u].sum, v);
			if ((tree[u].sum = 0) and (v = -1)) or ((tree[u].sum = 1) and (v = 1)) then
				inc(ans, tree[u].r - tree[u].p);
		end
		else
		begin
			Change(u shl 1, a, tree[u].m, v);
			Change(u shl 1 + 1, tree[u].m, b, v);
		end;
		exit;
	end;
	if tree[u].cover then
    begin
		tree[u shl 1].sum := tree[u].sum;
		tree[u shl 1 + 1].sum := tree[u].sum;
    end;
	if b <= tree[u].m then
		Change(u shl 1, a, b, v)
	else
		if a >= tree[u].m then
			Change(u shl 1 + 1, a, b, v)
		else
		begin
			Change(u shl 1, a, tree[u].m, v);
			Change(u shl 1 + 1, tree[u].m, b, v);
		end;
	if (tree[u shl 1].cover) and (tree[u shl 1 + 1].cover) and (tree[u shl 1].sum = tree[u shl 1 + 1].sum) then
	begin
		tree[u].cover := true;
		tree[u].sum := tree[u shl 1].sum;
	end
	else
		tree[u].cover := false;
end;
 
Procedure Main;
var
	i:longint;
begin
	qsort(1, n, 0);
	tree[1].p := 0;
	tree[1].r := 20001;
	Build_tree(1);
	for i := 1 to n do
		Change(1, l[i][0], r[i][0], flag[i][0]);

	for i := 1 to n do
		Change(1, l[i][1], r[i][1], flag[i][1]);
	writeln(ans);
end;
