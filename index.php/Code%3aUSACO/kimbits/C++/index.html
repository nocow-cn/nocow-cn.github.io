#include<fstream>

int a[35][35],tot[35][35];
int main()
{
    int n;
    int l;
    long long x;
    ifstream fin;
    fin.open("kimbits.in");
    fin>>n>>l>>x;
    fin.close();
    int i,j;
    for (i=0;i<=n;i++)
    {
        a[i][0]=1;
        tot[i][0]=1;
    }
    for (i=0;i<=l;i++)
    {
        tot[0][i]=1;
    }
    for (i=1;i<=n;i++)
    {
        tot[i][0]=1;
        for (j=1;j<=l;j++)
        {
            a[i][j]=a[i-1][j]+a[i-1][j-1];
            tot[i][j]=a[i][j]+tot[i][j-1];
        }
    }
    j=l;
    int leave=x-1;
    ofstream fout;
    fout.open("kimbits.out");
    for (i=n-1;i>=0;i--)
    {
        //cout<<leave<<" ";
        if (leave>=tot[i][j])
        {
           leave-=tot[i][j];
           j--;
           fout<<1;
        }
        else
        {
            fout<<0;
        }
    }
    fout<<endl;
    fout.close();
    return 0;
}
