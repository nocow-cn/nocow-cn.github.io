const maxn=1000000000;
var
  n,p,c:longint;
  count:array[1..800] of longint;
  a,b:array[1..800,0..800] of longint;
  d:array[1..20000] of integer;
  v:array[1..800] of boolean;
  dist:array[1..800] of longint;
  head,tail:longint;
  ans:longint;
procedure init;
var
  i,j,x,y,z:longint;
begin
  read(n,p,c);
  fillchar(count,sizeof(count),0);
  for i := 1 to n do
    begin
      read(x);
      inc(count[x]);
    end;
  for i := 1 to p do
    for j := 1 to p do
      a[i,j] := maxn;
  for i := 1 to c do
    begin
      read(x,y,z);
      inc(b[x,0]); b[x,b[x,0]] := y; a[x,y] := z;
      inc(b[y,0]); b[y,b[y,0]] := x; a[y,x] := z;
    end;
end;
//==================================
procedure spfa(s:longint);
var
  i,now,sum:longint;
begin
  fillchar(d,sizeof(d),0);
  fillchar(v,sizeof(v),false);
  for i := 1 to p do dist[i] := maxn;
  dist[s] := 0; v[s] := true; d[1] := s;
  head := 1; tail := 1;
  while head<=tail do
    begin
      now := d[head];
      for i := 1 to b[now,0] do
        if dist[b[now,i]]>dist[now]+a[now,b[now,i]] then
          begin
            dist[b[now,i]] := dist[now]+a[now,b[now,i]];
            if not v[b[now,i]] then
              begin
                inc(tail);
                d[tail] := b[now,i];
                v[b[now,i]] := true;
              end;
          end;
      v[now] := false;
      inc(head);
    end;
  sum := 0;
  for i := 1 to p do
    if count[i]<>0 then
      inc(sum,dist[i]*count[i]);
  if ans>sum then ans := sum;
end;
//==================================
procedure main;
var
  i:longint;
begin
  ans := maxn;
  for i := 1 to p do
      spfa(i);
end;
//======================================
procedure print;
begin
  writeln(ans);
end;
//=======================================
begin
  assign(input,'butter.in');
  reset(input);
  assign(output,'butter.out');
  rewrite(output);
 
  init;
  main;
  print;
 
  close(input);
  close(output);
end.

program Butter;
var
  farm:array [1..800] of integer; //每个农场的奶牛数
  a:array [1..800,0..800] of integer; //a[i,0]表示与农场i连通的农场数量，a[i,j]表示与农场i连通的第j个农场
  b:array [1..800,1..800] of integer; //邻接矩阵
  queue:array [1..1600] of integer; //SPFA队列
  f:array [1..800] of boolean; //记录农场是否在队列中
  d:array [1..800,1..800] of longint; //农场间的最短路径
  n,p,c,i,j,k,x,y,z:integer;
  s,ans:int64;
 
procedure work(k:integer); //SPFA
  var
    head,tail,x,i:integer;
  begin
    for i:=1 to p do begin
	  d[k,i]:=maxlongint;
	  f[i]:=false;
	end;
	d[k,k]:=0;
	head:=0;tail:=1;
	queue[1]:=k;
	f[k]:=true;
	while head<>tail do begin
	  head:=head mod p+1;
	  x:=queue[head];
	  for i:=1 to a[x,0] do
	    if d[k,a[x,i]]>d[k,x]+b[x,a[x,i]] then begin //松弛
		  d[k,a[x,i]]:=d[k,x]+b[x,a[x,i]];
		  if not f[a[x,i]] then begin //入队
		    tail:=tail mod p+1;
			queue[tail]:=a[x,i];
			f[a[x,i]]:=true;
		  end;
		end;
	  f[x]:=false; //出队
	end;
  end;
 
begin
  assign(input,'butter.in');reset(input);
  assign(output,'butter.out');rewrite(output);
  readln(n,p,c);
  fillchar(farm,sizeof(farm),0);
  fillchar(a,sizeof(a),0);
  for i:=1 to n do begin
    readln(k);
	inc(farm[k]);
  end;
  for i:=1 to c do begin
    readln(x,y,z);
	inc(a[x,0]);
	a[x,a[x,0]]:=y;
	inc(a[y,0]);
	a[y,a[y,0]]:=x;
	b[x,y]:=z;
	b[y,x]:=z;
  end;
  close(input);
  fillchar(queue,sizeof(queue),0);
  fillchar(f,sizeof(f),false);
  fillchar(d,sizeof(d),0);
  for i:=1 to p do
    work(i);
  ans:=-1;
  for i:=1 to p do begin
    s:=0;
    for j:=1 to p do
	  s:=s+farm[j]*d[i,j]; //计算总长度
	if (ans=-1) or (s<ans)
	  then ans:=s;
  end;
  writeln(ans);
  close(output);
end.

  i:word;

  heap[empt].dis:=dis;
  heap[empt].pi:=pi;
  i:=empt;
  while (i>1) and (heap[i div 2].dis>heap[i].dis) do
  begin
    sw:=heap[i];
    heap[i]:=heap[i div 2];
    heap[i div 2]:=sw;
    i:=i div 2;
  end;
  while heap[empt].dis<>oo do
    inc(empt);
end;

var
  a,b:word;
begin
  a:=s*2;

  begin
    if s<empt then

       list:array[1..2900] of longint;

procedure spfa_slf_lll(u:longint); 

       var
               fi,en,k,i,j,su,count:longint;
               p:bi;
       begin
               su:=0;count:=1;
               fillchar(f,sizeof(f),0);
               fillchar(d,sizeof(d),255);
               d[u]:=0;fi:=0;en:=1;
               list[1]:=u;f[u]:=true;
               while fi<>en do
               begin
                       fi:=fi mod mo+1;//
                       while d[list[fi]]>su div count do
                       begin
                               en:=en mod mo+1;
                               list[en]:=list[fi];
                               fi:=fi mod mo+1;
                       end;//LLL优化
                       j:=list[fi];
                       p:=bian[list[fi]];
                       dec(count);
                       dec(su,d[j]);
                       while p<>nil do
                       begin
                               i:=p^.dq;
                               if (d[j]+p^.qz<d[i])or(d[i]=-1) then
                               begin
                                       d[i]:=d[j]+p^.qz;
                                       if not f[i] then begin
                                       inc(count);
                                       inc(su,d[i]);//
                                       if d[i]<d[list[fi mod mo+1]] then
                                       begin
                                               f[i]:=true;
                                               list[fi]:=i;
                                               fi:=(fi+mo-2) mod mo+1;
                                       end//SLF优化
                                       else
                                       begin
                                               f[i]:=true;
                                               en:=en mod mo+1;
                                               list[en]:=i;
                                       end;
                                       end;
                               end;
                               p:=p^.next;
                       end;
                       f[j]:=false;
               end;
               for i:=1 to ptt do
                       if d[i]<>-1 then inc(sum[i],d[i]) else
                               inc(sum[i],maxlongint div 2);
       end;
