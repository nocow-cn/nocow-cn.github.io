program snail_sd;
type
  direction=(up,down,left,right);
const
  maxn=120;
  dx:array[direction]of shortint=(-1,1,0,0);
  dy:array[direction]of shortint=(0,0,-1,1);
var
  fin,fout:text;
  map:array[0..maxn+1,0..maxn+1]of char;
  n,b,i,j:byte;
  now,best:integer;
  c:char;
procedure search(x0,y0:byte;d:direction);
  var
    x,y:byte;
  begin
    x:=x0;y:=y0;
    if map[x+dx[d],y+dy[d]]<>'.' then exit;
    repeat
    inc(x,dx[d]);inc(y,dy[d]);
      map[x,y]:='x';
      inc(now);
    until map[x+dx[d],y+dy[d]]<>'.';
    if now>best then best:=now;
    if map[x+dx[d],y+dy[d]]<>'x' then begin
      search(x,y,up);
      search(x,y,down);
      search(x,y,left);
      search(x,y,right);
    end;
    while (x<>x0) or (y<>y0) do begin
      dec(now);
      map[x,y]:='.';
      dec(x,dx[d]);dec(y,dy[d]);
    end;
  end;
begin
  assign(fin,'snail.in');
  reset(fin);
  readln(fin,n,b);
  for i:=1 to n do begin
    map[0,i]:='#';
    map[n+1,i]:='#';
    map[i,0]:='#';
    map[i,n+1]:='#';
  end;
  for i:=1 to n do
    for j:=1 to n do
      map[i,j]:='.';
  for i:=1 to b do begin
    read(fin,c);
    readln(fin,j);
    map[j,ord(c)-64]:='#';
  end;
  close(fin);
 
  best:=1;now:=1;
  map[1,1]:='x';
  search(1,1,down);
  search(1,1,right);
 
  assign(fout,'snail.out');
  rewrite(fout);
  writeln(fout,best);
  close(fout);
end.

program gu;
type
  xy=record x,y:integer; end;
var
  pic:array[0..121,0..121]of integer;  //pic=0 表示空；pic=1 表示走过；pic=2 表示障碍。
  walk:array[0..3]of xy;
  n,max:longint;
procedure fill;
  begin
    walk[0].y:= 1;
    walk[1].x:= 1;
    walk[2].y:=-1;
    walk[3].x:=-1;
  end;
procedure init;
  var
    m,i,x,y,f:longint;
    s:string;
  begin
    readln(n,m);
    for i:=1 to m do
    begin
      readln(s);
      y:=ord(s[1])-64;
      delete(s,1,1);
      val(s,x,f);
      pic[x,y]:=2;;
    end;
    for i:=0 to n+1 do
    begin
      pic[0,i]:=2;
      pic[i,0]:=2;
      pic[n+1,i]:=2;
      pic[i,n+1]:=2;
    end;
  end;
procedure find(x,y,p,how:longint);
  var
    how2:longint;  //how2=-1 时不转；how2<>-1 时有how和how2两种情况
  begin
    how2:=-1;
    if pic[x,y]=1 then exit;
    pic[x,y]:=1;
    inc(p);
    if p>max then max:=p;
    if pic[x+walk[how].x,y+walk[how].y]=2 then
    begin
      how2:=(how+1)mod 4;
      how:=(how+3)mod 4;
    end;
    if pic[x+walk[how].x,y+walk[how].y]<>2 then
       find(x+walk[how].x,y+walk[how].y,p,how);
    if how2<>-1 then
    begin
      how:=how2;
      if pic[x+walk[how].x,y+walk[how].y]<>2 then
         find(x+walk[how].x,y+walk[how].y,p,how);
    end;
    pic[x,y]:=0;
  end;
procedure doit;
  begin
    find(1,1,0,0);
    find(1,1,0,1);
    writeln(max);
  end;

  assign(input,'snail.in');
  assign(output,'snail.out');
  reset(input);
  rewrite(output);
  fill;
  init;
  doit;
  close(input);
  close(output);

program snail;
const inf='snail.in';
      ouf='snail.out';
      bu:array[0..1,1..2,1..2]of longint=
      (((1,0),(-1,0)),((0,1),(0,-1)));
var
   n,m,ans,i,j,y:longint;
   map:array[0..121,0..121]of char;
   ch:char;
 
procedure sou(x,y,dir,dis:longint);  {（坐标X,Y；方向（竖直或水平）；已走的距离）}
var
   i,now_x,now_y,now_dis,xl,xr,yl,yr:longint;
 
   procedure stepback;
 
   begin
     now_x:=now_x-bu[dir,i,1];
     now_y:=now_y-bu[dir,i,2];
   end;
 
   procedure recover;
   var
      j1,j2:longint;
   begin
     if x>now_x then begin xl:=now_x;xr:=x;end
                else begin xl:=x;xr:=now_x;end;
     if y>now_y then begin yl:=now_y;yr:=y;end
                else begin yl:=y;yr:=now_y;end;
     for j1:=xl to xr do
       for j2:=yl to yr do
         map[j1,j2]:=' ';
   end;
begin
  for i:=1 to 2 do
  begin
    now_dis:=dis;
    now_x:=x+bu[dir,i,1];
    now_y:=y+bu[dir,i,2];
    if (now_x>0)and(now_x<=N)
    and(now_y>0)and(now_y<=N)
    and(map[now_x,now_y]<>'#') then
    begin
      while (map[now_x,now_y]<>'#')and(map[now_x,now_y]<>'$')
            and(now_x>0)and(now_x<=N)and(now_y>0)and(now_y<=N) do
      begin
        map[now_x-bu[dir,i,1],now_y-bu[dir,i,2]]:='$';
        now_x:=now_x+bu[dir,i,1];
        now_y:=now_y+bu[dir,i,2];
        inc(now_dis);
      end;
      if now_dis>ans then ans:=now_dis;
      if map[now_x,now_y]='$' then
      begin
        stepback;
        recover;
        CONTINUE;
      end;
 
      stepback;
 
      if (now_x<>x)or(now_y<>y) then
        sou(now_x,now_y,1-dir,now_dis);
 
      recover;
    end;
  end;
end;
 
BEGIN
  assign(input,inf);reset(input);
  assign(output,ouf);rewrite(output);
  readln(n,m);
  for i:=1 to n do
    for j:=1 to n do
      map[i,j]:=' ';
  for i:=1 to m do
  begin
    readln(ch,y);
    map[y,ord(ch)-ord('A')+1]:='#';
  end;
  sou(1,1,0,1);
  sou(1,1,1,1);
  writeln(ans);
  close(input);close(output);
