  exit(false);
end;
function find(x,p:longint):boolean;
begin
  if p>length(t) then exit(true);
  if a[x,t[p]]>0 then find:=find(a[x,t[p]],p+1)
                 else exit(false);
end;
procedure main;
begin
  m:=0; fillchar(ans,sizeof(ans),0); val:=0; y:=0;
  for i:=3 to l do
   for j:=p[i-1]+1 to p[i] do
   begin
    t:=s[j];
    if not find(1,1) then continue;
    now:=0;
    for k:=1 to i do
     now:=score[s[j,k]]+now;
    if now>val then
     begin
       val:=now; m:=0; fillchar(ans,sizeof(ans),0);
     end;
    if now=val then
     begin
       inc(m); ans[m]:=s[j];
     end;
   end;
  for i:=3 to l do
   for k:=p[i-1]+1 to p[i] do
   begin
    if s[k]='mam' then
     t:=t;
    t:=s[k]; if not find(1,1) then continue;
   for j:=3 to l-i do
     for y:=p[j-1]+1 to p[j] do
     if min(s[k],s[y]) then
     begin
       t:=s[k]+s[y];
       if not find(1,1) then continue;
       now:=0;
       for x:=1 to i do
        inc(now,score[s[k,x]]);
       for x:=1 to j do
        inc(now,score[s[y,x]]);
      if now>val then
      begin
        val:=now; m:=0; fillchar(ans,sizeof(ans),0);
      end;
     if now=val then
      begin
        inc(m);
        ans[m]:=s[k]+' '+s[y];
      end;
     end;
  end;
end;
procedure qsort(h,r:longint);
var
  i,j:longint;
  k,t:string;
begin
  i:=h; j:=r; k:=ans[random(j-i)+i];
  repeat
    while min(ans[i],k) do inc(i);
    while min(k,ans[j]) do dec(j);
    if i<=j then
     begin
       t:=ans[i]; ans[i]:=ans[j]; ans[j]:=t;
       inc(i); dec(j);
     end;
  until i>j;
  if i<r then qsort(i,r);
  if h<j then qsort(h,j);
end;
