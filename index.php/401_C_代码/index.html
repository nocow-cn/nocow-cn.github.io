#include<stdio.h>
#include<string.h>
char s[1000],k[100]={"112S3E4 5Z6 7 889               AAB C D E3F G HHIIJLK LJMMN OOP Q R S2TTUUVVWWXXYYZ5"};
int l,r,i,j,a,b,c;
int main()
{
    i=0;
    while (scanf("%s",s)==1)
    {
          //if (i != 0) { printf("\n"); }//printf("\n"); 
          l = 0;
          r = strlen(s)-1;
          b = 0; c = 0;
          if ( ((r-l+1) % 2)==1 )
          {
               while (l <= r)
               {
                     a = (s[l]-'1')*2;
                     if (s[l] == s[r]) b++;
                     if (s[r] == k[a+1]) c++;
                     l++;
                     r--;
               }
               j = strlen(s);
               if (b == ((j+1)/2))
               {
                     if (c == ((j+1)/2)) printf("%s -- is a mirrored palindrome.\n\n",s);
                     else printf("%s -- is a regular palindrome.\n\n",s);  
               }
               else if (c == ((j+1)/2)) printf("%s -- is a mirrored string.\n\n",s);
                    else printf("%s -- is not a palindrome.\n\n",s); 
          }
          else
          {
              while (l < r)
              {
                    a = (s[l]-'1')*2;
                    if (s[l] == s[r]) b++;
                    if (s[r] == k[a+1]) c++;
                    l++;
                    r--;
              }
			  j = strlen(s);
              if (b == (j/2))
              {
                    if (c == (j/2)) printf("%s -- is a mirrored palindrome.\n\n",s);
                    else printf("%s -- is a regular palindrome.\n\n",s);  
              }
              else if (c == (j/2)) printf("%s -- is a mirrored string.\n\n",s);
                   else printf("%s -- is not a palindrome.\n\n",s);
          }       
          i++;     
    }  
    return 0;
}
