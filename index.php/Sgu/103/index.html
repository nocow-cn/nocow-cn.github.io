void Init()
{
	fin >> s >> t;
	fin >> n >> m ;
	for(int i(1); i<= n; ++ i)
	{
		char chr;
		int t1, t2, t3;
 
		//R[i].Clr[0]= false; R[i].Clr[1]= true;
		fin >> chr >> t1 >> t2 >> t3;
		//R[i].Ini= (chr == 'P');
		if(chr == 'P')
			R[i].KeepIni= t2+ t3- t1;
		else 
			R[i].KeepIni= t2- t1;
		R[i].Keep[0]= t2;
		R[i].Keep[1]= t2+ t3;
	}
	for(int i(1); i<= m; ++ i)
	{
		int t1, t2, val;
		fin >> t1 >> t2 >> val;
		G[t1][++G[t1][0]]= t2;
		V[t1][++V[t1][0]]= val;
		G[t2][++G[t2][0]]= t1;
		V[t2][++V[t2][0]]= val;
	}
}
 
int Calc(int a, int b, int t)
{
	int ta, tb, Tmp, Time;
 
	ta= (t+ R[a].KeepIni) % R[a].Keep[1];
	tb= (t+ R[b].KeepIni) % R[b].Keep[1];
	Time= 0;
 
	for(int i= 0; i< 10; ++ i)
		if(( ta < R[a].Keep[0] ) == (tb <  R[b].Keep[0] ))
		{
			return Time;
		}else
		{
			if(ta < R[a].Keep[0])
				Tmp= R[a].Keep[0]- ta;
			else
				Tmp= R[a].Keep[1]- ta;
			if(tb < R[b].Keep[0])
				Tmp= min(Tmp, R[b].Keep[0]- tb);
			else
				Tmp= min(Tmp, R[b].Keep[1]- tb);
			Time+= Tmp;
			ta= (ta + Tmp) % R[a].Keep[1];
			tb= (tb + Tmp) % R[b].Keep[1];
		}
	return -1;
}
