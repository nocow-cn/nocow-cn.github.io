var c:array[1..10] of integer;
    n,max:longint;
 
function calc(a,b:integer):longint;
  var r:integer;
  begin
    calc:=a*b;
    r:=a mod b;
    while r<>0 do
      begin
        a:=b;
        b:=r;
        r:=a mod b;
      end;
    calc:=calc div b;
  end;
 
procedure init;
  var i,j,temp:integer;
  begin
    readln(n);
    for i:=1 to n do
      readln(c[i]);
    for i:=1 to n-1 do
      for j:=i+1 to n do
        if c[i]<c[j] then begin temp:=c[i]; c[i]:=c[j]; c[j]:=temp; end;
    if n>=2 then max:=calc(c[1],c[2]) else max:=calc(c[1],1);
  end;
 
procedure work;
  var f:array[0..80000] of boolean;
      i:integer;
      k,ans:longint;
  begin
    fillchar(f,sizeof(f),false);
    f[0]:=true;
    k:=0;
    while k<=max do
      begin
        if f[k] then
          for i:=1 to n do
            f[k+c[i]]:=true;
        k:=k+1;
        if not(f[k]) then ans:=k;
      end;
    if ans<=max then writeln(ans) else writeln(0);
  end;

  close(input);

end.
var n,i,j,d,t,a,b,k:longint;
    s:array[1..10]of integer;
    w:array[-1..255]of longint;
function gcd(a,b:integer):integer;//gcd求最大公约数
    var t:integer;
    begin
        if a>b then begin
            t:=a;
            a:=b;
            b:=t
        end;
        while a>0 do begin
            t:=b mod a;
            b:=a;
            a:=t
        end;
        gcd:=b
    end;
begin
    assign(input,'nuggets.in');reset(input);
    assign(output,'nuggets.out');rewrite(output);
    readln(n);
    for i:=1 to n do
        readln(s[i]);
    d:=s[1];
    for i:=2 to n do
        d:=gcd(d,s[i]);
    if d>1 then begin
        writeln(0);
        close(output);
        halt
    end;
    //排序
    for i:=1 to n-1 do
        for j:=i+1 to n do
            if s[i]>s[j] then begin
                t:=s[i];
                s[i]:=s[j];
                s[j]:=t
            end;
    fillchar(w,sizeof(w),255);
    w[0]:=0;
    for i:=2 to n do
        for j:=0 to s[1]-1 do begin
            k:=j;
            while k<=j do begin
                if w[j]<>-1 then begin
                    a:=k+s[i];
                    b:=(a div s[1])+w[k];
                    a:=a mod s[1];
                    if (w[a]>b)or(w[a]=-1) then begin
                        w[a]:=b;
                        k:=a end
                    else
                        break end
                else
                    break
            end
        end;
    a:=0;
    for i:=s[1]-1 downto 1 do
        if w[i]>w[a] then
            a:=i;
    w[0]:=1;
    writeln((w[a]-1)*s[1]+a);
    close(output)
end.

var

  opt     :array[1..10] of integer;
 
FUNCTION GCD(p,q:integer):integer;
BEGIN
  if q=0 then
   GCD:=P else
   GCD:=GCD(q,p mod q);
END;
 
PROCEDURE FINDANS;
var
  can     :array[1..66000] of boolean;
  max   :word;
BEGIN
  fillchar(can,sizeof(can),false);
  for i:=1 to n do
    can[opt[i]]:=true;
  max:=opt[n]*(opt[1]-2)-opt[1];
  i:=opt[n];
  while i<=max do begin
    if can[i-1] then begin
      inc(t);
      if t=opt[n] then begin
        ans:=i-t;
        exit;
      end;
    end else t:=1;
    for j:=1 to n do
      can[i+opt[j]]:=true;
    while not can[i+1] do inc(i);
    inc(i);
  end;
  ans:=max+opt[n];
  while can[ans] do
    dec(ans);
END;
 
BEGIN
  Assign(input,'nuggets.in'); Reset(Input);
  readln(n);
  for i:=1 to n do begin
    readln(opt[i]);
    for t:=1 to i-1 do
      if opt[t]<opt[i] then begin
        d:=opt[t];  opt[t]:=opt[i]; opt[i]:=d;
      end;
  end;
  Close(Input);
 
  d:=opt[n];
  for i:=1 to n-1 do
    d:=GCD(d,opt[i]);
  if d>1 then
    ans:=0 else
    FINDANS;
 
  Assign(Output,'nuggets.out'); Rewrite(Output);
  writeln(ans);
  Close(Output);
END.

var
  n,maxn,ans,i,j:longint;
  a:array[0..10]of longint;
  f:Array[0..2000000]of boolean;
 
begin
  assign(input,'nuggets.in');reset(input);
  assign(output,'nuggets.out');rewrite(output);
  readln(n);
  for i:=1 to n do
  begin
    readln(a[i]);
    if a[i]>maxn then maxn:=a[i];
  end;
  ans:=0;
  i:=0;
  f[0]:=true;
  while true do
  begin
    inc(i);
    for j:=1 to n do
      if (i>=a[j])and(f[i-a[j]]) then
      begin
        f[i]:=true;
        break;
      end;
    if not f[i] then
      ans:=i;
      if i-ans>maxn then break;
    if i>200000 then begin ans:=0; break; end;
  end;
  writeln(ans);
  close(output);
end.

begin
   if b=0 then exit(a) else
   exit(gcd(b,a mod b));

var a:array[0..65536] of longint;
    b:array[1..10] of integer;
    j,k,l,n,m,i,o,p:longint;
begin
    assign(input,'nuggets.in');
    reset(input);
    assign(output,'nuggets.out');
    rewrite(output);
    readln(n);
    for i:=1 to n do
    readln(b[i]);
    o:=0;
    for i:=1 to n do
    for j:=1 to n do
    begin
     if i<>j then
      if gcd(b[i],b[j])=1 then inc(o);
    end;
    if o=0 then
     begin
      writeln(0);
      halt;
     end;
    a[0]:=1;
    for i:=1 to n do
    begin
     for j:=0 to 65536 do
     begin
      if (a[j]<>0)and(j+b[i]<=65536) then
       inc(a[j+b[i]]);
     end;
    end;
    for i:=65536 downto 1 do
    if a[i]=0 then
    begin
     writeln(i);
