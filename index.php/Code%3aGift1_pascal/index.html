type
    data=record
            name:string[14];
            intend_to_give:longint;
            account:longint
            end;
var
    a:array[1..10]of data;
    giver,receiver:string[14];
    i,j,np,np_receive,g,r:byte;
    plus,money_to_give,remain:longint;
    fin,fout:text;
function check(nm:string):byte;
var
    t:byte;
begin
    for t:=1 to np do
        if a[t].name=nm then begin
            check:=t;
            break;
        end;
end;
 
begin
    assign(fin,'gift1.in');
    reset(fin);
    assign(fout,'gift1.out');
    rewrite(fout);
    for i:=1 to 10 do           {存储数组初始化}
        with a[i] do begin
            name:='';
            account:=0;
        end;
    readln(fin,np);
    for i:=1 to np do
        readln(fin,a[i].name);
    for i:=1 to np do begin
        readln(fin,giver);
        g:=check(giver);
        readln(fin,money_to_give,np_receive);
 
        if money_to_give=0 then
            if np_receive<>0 then begin
                for j:=1 to np_receive do readln(fin);
                continue;
            end
            else continue;
        if np_receive=0 then begin
            a[g].account:=a[g].account+money_to_give;
            continue;
            end;
 
        remain:=money_to_give mod np_receive;
        a[g].account:=a[g].account+remain;
        a[g].intend_to_give:=money_to_give;
        plus:=money_to_give div np_receive;
        for j:=1 to np_receive do begin
            readln(fin,receiver);
            r:=check(receiver);
            a[r].account:=a[r].account+plus;
        end;
    end;
    for i:=1 to np do
        with a[i] do
            writeln(fout,name,' ',account-intend_to_give);
    close(fin);
    close(fout)
end.

var
  n, i, j, k, t, p, q: longint;
  name: array[1..10] of string;
  a, b: array[1..10] of longint;
  s: string;
begin
 
  assign(input,'gift1.in');
  reset(input);
  assign(output,'gift1.out');
  rewrite(output);
 
 
  readln(n);
  for i := 1 to n do readln(name[i]);
  for i := 1 to n do begin
    readln(s);
    j := 1;
    while name[j] <> s do inc(j);
    readln(a[j],t);
    if t = 0 then begin
      b[j] := b[j] + a[j];
      a[j] := 0;
    end else begin
      p := a[j] div t;
      q := a[j] mod t;
      inc(b[j],q);
      for k := 1 to t do begin
        readln(s);
        j := 1 ;
        while name[j] <> s do inc(j);
        inc(b[j],p);
      end; //for k
    end;//else
 
  end; //for i
  for i := 1 to n do a[i] := b[i]-a[i];
  for i := 1 to n do writeln(name[i],' ',a[i]);
 
  close(output);
  close(input);
end.
