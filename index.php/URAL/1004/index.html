#include <iostream>
#include <stdio.h>
using namespace std;
 
#define M 10000000
int dist[105][105], data[105][105];
int n, t, path[105], f[105][105], mins;
void extend_floyd()
{
    int i, j, k, tmp, st;
    mins = M;
    for(k = 1; k <= n; k ++)
    {
        for(i = 1; i < k; i ++)
        {
            for(j = i+1; j < k; j ++)
            {
                tmp = data[i][j] + dist[j][k] + dist[k][i];
                if(tmp < mins)
                {
                    mins = tmp;
                    st = j;
                    t = 0;
                    while(st != i)
                    {
                        path[t++] = st;
                        st = f[i][st];
                    }
                    path[t++] = i;
                    path[t++] = k;
                }
            }
        }
        for(i = 1; i <= n; i ++)
        {
            for(j = 1; j <= n; j ++)
            {
                tmp = data[i][k] + data[k][j];
                if(tmp < data[i][j])
                {
                    data[i][j] = tmp;
                    f[i][j] = f[k][j];
                }
            }
        }
    }
}
