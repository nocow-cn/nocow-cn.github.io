	int k = 0;

	cin >> n;
	while(cin >> s)
	{
		if(s.length() == n && count(s)) goto end;
		if(s.length() == n)
		for(int i = 1 ; i <= n ; i++)
		if(s[i - 1] == '1')
		{
			s[i - 1] = '0';
			if(count(s)) goto end;
			s[i - 1] = '1';
		}
		if(s.length() < n)
		for(int i = 1 ; i <= n ; i++)
		{
			s.insert(i - 1, "1");
			if(count(s)) goto end;
			s.erase(i - 1, 1);
			s.insert(i - 1, "0");
			if(count(s)) goto end;
			s.erase(i - 1, 1);

		if(s.length() > n)
		for(int i = 1 ; i <= n ; i++)
		{
			string tmp = s.substr(i - 1, 1);
			s.erase(i - 1, 1);
			if(count(s)) goto end;
			s.insert(i - 1, tmp);
		}
		end:
			cout << s << endl;
	}
}
