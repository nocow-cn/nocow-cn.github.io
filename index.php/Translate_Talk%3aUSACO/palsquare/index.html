Translate Talk:USACO/palsquare

来自"NOCOW"

跳转到: 导航, 搜索

描述

回文数是指从左向右念和从右像做念都一样的数。如12321就是一个典型的回文数。

给定一个进制B(2<=B<=20十进制)，输出所有的大于等于1小于等于300且它的平方用B进制表示时是回文数的数。用’A’,’B’……表示10，11等等。 [编辑] 格式

PROGRAM NAME: palsquare

INPUT FORMAT:

(file palsquare.in)

共一行，一个单独的整数B(B用十进制表示)。

OUTPUT FORMAT:

(file palsquare.out)

每行两个数字，第二个数是第一个数的平方，且第二个数是回文数。(注意:这两个数都应该在B那个进制下) program paslquare(input,output); 

 const
   inf='paslquare.in';
   outf='paslquare.out';
 var
   i,j,t,m,b:integer;
   f:boolean;
   s:array[1..20]of integer;
   a:array[1..10]of integer;
 procedure search;
   begin
     repeat
       m:=m+1;
     until a[m]<(m-1);
    end;
 procedure multiply;
   var
     j,k:byte;
   begin
     for j:=m downto 1 do
       for k:=m downto 1 do
         begin
           s[j+k]:=s[j+k]+(a[j]*a[k])div b;
           if k<>m
             then s[j+k]:=s[j+k]+(a[j]*a[k+1])mod b;
           if s[j+k]>(b-1)
             then
               begin
                 s[j+k]:=s[j+k]mod b;
                 s[i+k+1]:=s[j+k+1]+s[j+k]div b
               end;
         end;
     repeat
       t:=t+1;
     until s[t]=0;
     t:=t-1;
   end;
 begin
   assign(input,inf);
   reset(input);
   assign(output,outf);
   reset(output);
   read(input,b);
   for i:=1 to 10 do
     a[i]:=0;
   for i:=1 to 20 do
     s[i]:=0;
   m:=0;
   t:=0;
   for i:=1 to 300 do
     begin
       search;
       a[m]:=a[m]+1;
       multiply;
       f:=true;
       for j:=1 to t do
         if s[j]<>s[t-j+1]
           then f:=false;
       if f=true
         then   begin
                  writeln(output);
                  for j:=1 to m do
                    write(output,a[j]);
                  write(output,' ');
                  for j:=1 to t do
                    write(output,s[j]);
                 end;
       m:=0;
       t:=0;
       for j:=1 to 20 do
         s[j]:=0;
     end;
   close(input);
   close(output);

end.
