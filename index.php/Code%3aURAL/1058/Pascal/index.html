// n^3 algo. for ural 1058, Accepted 0.015s
const err=1e-10; maxn=50;
type po=record x,y:double; end;
     plg=record
        n:longint;
        a:array[0..maxn]of po;
     end;
var i,j,k,m,n:longint;
    pg:plg;p:po;
    q,r,s,ss:double;
function chaji(var p0,p1,p2:po):double;
begin
    chaji:=(p1.x-p0.x)*(p2.y-p0.y)-(p2.x-p0.x)*(p1.y-p0.y);
end;
function area(var p:plg):double;
var i:longint;
    s:double; opo:po;
begin
    s:=0; opo.x:=0; opo.y:=0;
    for i:=0 to p.n-1 do
       s:=s+chaji(opo,p.a[i],p.a[i+1]);
    area:=s/2;
end;
function angle(var p,q:po):double;
var a:double;
begin
    if abs(p.x-q.x)<err then a:=pi/2 else a:=arctan((p.y-q.y)/(p.x-q.x));
    if abs(a)<pi/4 then begin
       if q.x<p.x then a:=a+pi;
    end else begin
       if q.y<p.y then a:=-abs(a) else a:=abs(a);
    end;
    angle:=a;
end;
function dist(var p,q:po):double;
begin
    dist:=sqrt(sqr(p.x-q.x)+sqr(p.y-q.y));
end;
function calck(var p1,p2,q1,q2:po):double;
var a,b,aa,bb:double;
begin
    a:=angle(p1,p2);
    b:=angle(q1,q2);
    aa:=(a+b)/2;
    if abs(cos(aa))>err then bb:=sin(aa)/cos(aa) else bb:=9e99;
    calck:=bb;
end;
function workp(var pg:plg; x:longint):double;
var i,ii:longint;
    s,r,l,ll:double;
    p:po;
begin
    s:=0;
    for ii:=x+1 to x-1+pg.n do begin
       i:=ii mod pg.n;
       r:=abs(chaji(pg.a[x],pg.a[i],pg.a[i+1])/2);
       s:=s+r;
       if s>0.5*ss then break;
    end;
    ll:=(s-0.5*ss)/r; // a[i] %
    p.x:=ll*pg.a[i].x+(1-ll)*pg.a[i+1].x;
    p.y:=ll*pg.a[i].y+(1-ll)*pg.a[i+1].y;
    l:=dist(pg.a[x],p);
    workp:=l;
end;
function calcb(k:double;p:po):double;inline;
begin
    calcb:=p.y-k*p.x;
end;
function upline(p:po;k,b:double):longint;//1-up 0-on -1-down
begin
    k:=k*p.x+b;
    if abs(p.y-k)<err then upline:=0 else if p.y>k then upline:=1
      else upline:=-1;
end;
procedure calcpo(var p,q:po;k,b:double;var a:po);
var x,y,z:double;
begin
    x:=calcb(k,p); y:=calcb(k,q);
    z:=(b-x)/(y-x);
    a.x:=z*q.x+(1-z)*p.x;
    a.y:=z*q.y+(1-z)*p.y;
end;
procedure calclr(var p1,p2,q1,q2:po;k:double;var l,r:double);
var i,j:longint; t:double;
    b:array[1..4]of double;
begin
    b[1]:=calcb(k,p1); b[2]:=calcb(k,p2);
    b[3]:=calcb(k,q1); b[4]:=calcb(k,q2);
    for i:=1 to 3 do for j:=i+1 to 4 do if b[i]>b[j] then begin
       t:=b[i]; b[i]:=b[j]; b[j]:=t;
    end;
    t:=(b[2]+b[3])/2;
    l:=b[2]; r:=b[3];
    if (upline(p1,k,t)*upline(p2,k,t)<>-1)
      or(upline(q1,k,t)*upline(q2,k,t)<>-1) then begin
         l:=1; r:=-1;
      end;
end;
procedure worke;
var i,j,k:longint;
    q,s,t,u,s1,s2:double;
    p1,p2:plg; pa,pb,pc,pd,pt:po;
begin
    for i:=0 to pg.n-2 do for j:=i+1 to pg.n-1 do begin
       q:=calck(pg.a[i],pg.a[i+1],pg.a[j],pg.a[j+1]);
       if abs(q)>1.6 then continue;
       calclr(pg.a[i],pg.a[i+1],pg.a[j],pg.a[j+1],q,s,t);
 
       if s>t-err then continue;
 
       calcpo(pg.a[i],pg.a[i+1],q,s,pa);
       calcpo(pg.a[i],pg.a[i+1],q,t,pb);
       if dist(pa,pg.a[i])>dist(pb,pg.a[i]) then begin
          pt:=pa; pa:=pb; pb:=pt;
       end;
       calcpo(pg.a[j],pg.a[j+1],q,s,pc);
       calcpo(pg.a[j],pg.a[j+1],q,t,pd);
       if dist(pc,pg.a[j])>dist(pd,pg.a[j]) then begin
          pt:=pc; pc:=pd; pd:=pt;
       end;
 
       p2.n:=j-i+2; p1.n:=pg.n-p2.n+4;
       for k:=0 to i do p1.a[k]:=pg.a[k];
       p1.a[i+1]:=pa; p1.a[i+2]:=pd;
       for k:=i+3 to p1.n-1 do p1.a[k]:=pg.a[k+j-i-2];
       p1.a[p1.n]:=p1.a[0];
 
       p2.a[0]:=pb; p2.a[p2.n]:=pb;
       for k:=1 to p2.n-2 do p2.a[k]:=pg.a[k+i];
       p2.a[p2.n-1]:=pc;
 
       s1:=0.5*ss-abs(area(p1)); s2:=0.5*ss-abs(area(p2));
       if (s1<-err)or(s2<-err) then continue;
       if s1<err then s1:=0;
       if s2<err then s:=dist(pb,pc) else begin
          u:=-s1/s2;
          s:=dist(pa,pd);
          t:=dist(pc,pb);
          s:=(s*s-u*t*t)/(1-u);
          if s<0 then continue;
          s:=sqrt(s);
       end;
       if s<r then r:=s;
    end;
end;
